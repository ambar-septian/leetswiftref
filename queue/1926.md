```swift
class Solution {
  func nearestExit(_ maze: [[Character]], _ entrance: [Int]) -> Int {
    guard !maze.isEmpty else { return -1 }

    var maze = maze
    let dirs = [(0, 1), (0, -1), (1, 0), (-1, 0)]
    let rows = maze.count
    let cols = maze[0].count

    // r * cols + c
    let entranceIndex = entrance[0] * cols + entrance[1]
    var queue = [entranceIndex]
    var results = 0

    while !queue.isEmpty {
      let count = queue.count
      for _ in 0..<count {
        let evidict = queue.removeFirst()

        let r = evidict / cols
        let c = evidict % cols

        for dir in dirs {
          let rD = r + dir.0
          let cD = c + dir.1

          if rD < 0 || rD >= rows || cD < 0 || cD >= cols {

            if entranceIndex != evidict {
              return results
            }

          } else {
            if maze[rD][cD] == "." {
              let index = rD * cols + cD
              maze[rD][cD] = "+"
              queue.append(index)
            } else {
              continue
            }
          }
        }
      }

      results += 1

    }

    return -1

    /*
Problem: Nearest Exit from Entrance in Maze (BFS)
Time Complexity: O(M * N)
    - M: Number of rows, N: Number of columns.
    - Each cell is visited and processed once.
    - (Assumes O(1) dequeue; Swift's removeFirst() is O(K)).
Auxiliary Space: O(M * N)
    - O(M * N) for the BFS queue in the worst case.
    - O(M * N) for the local grid copy in Swift (Copy-on-Write).
*/
  }
}
```
