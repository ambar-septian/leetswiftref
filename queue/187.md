```swift
class Solution {
  func findRepeatedDnaSequences(_ s: String) -> [String] {
    guard s.count >= 10 else { return [] }

    var seen = Set<Substring>()
    var result = Set<Substring>()

    let startIndex = s.startIndex
    var left = startIndex
    var right = s.index(startIndex, offsetBy: 9)

    while right < s.endIndex {
      let sub = s[left...right]

      if seen.contains(sub) {
        result.insert(sub)
      } else {
        seen.insert(sub)
      }

      left = s.index(after: left)
      right = s.index(after: right)
    }

    return result.map(String.init)
  }
}
```
