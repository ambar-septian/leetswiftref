```swift
class Solution {
  func maxVowels(_ s: String, _ k: Int) -> Int {
    let vowels: Set<Character> = Set(arrayLiteral: "a", "e", "i", "u", "o")
    let chars = Array(s)
    let count = chars.count
    var result = 0
    var curr = 0

    for i in 0..<count {
      if vowels.contains(chars[i]) {
        curr += 1
      }

      if i >= k {
        if vowels.contains(chars[i - k]) {
          curr -= 1
        }
      }

      result = max(result, curr)
    }

    return result
  }
}
```
