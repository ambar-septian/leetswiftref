```swift
class Solution {
  func search(_ nums: [Int], _ target: Int) -> Int {
    var low = 0
    var height = nums.count

    while low <= height {
      let middle = (low + height) / 2
      guard nums.count > middle else { break }

      if nums[middle] == target {
        return middle
      } else if nums[middle] < target {
        low = middle + 1
      } else if nums[middle] > target {
        height = middle - 1
      }
    }

    return -1
  }
}
```
