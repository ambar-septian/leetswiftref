```swift
class Solution {
  func threeConsecutiveOdds(_ arr: [Int]) -> Bool {
    var current = 0

    for i in 0..<arr.count {
      if !(arr[i] % 2 == 0) {
        current += 1
      } else if current < 3 {
        current = 0
      }
    }

    return current >= 3
  }
}
```
