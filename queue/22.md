```swift
class Solution {
  func generateParenthesis(_ n: Int) -> [String] {
    var results = [String]()
    var path = [Character]()
    path.reserveCapacity(n * 2)

    func dfs(_ o: Int, _ c: Int) {
      if c == n && o == n {
        let s = String(path)
        results.append(s)
        return
      }

      if o < n {
        path.append("(")
        dfs(o + 1, c)
        path.removeLast()
      }

      if c < o {
        path.append(")")
        dfs(o, c + 1)
        path.removeLast()
      }
    }
    
    dfs(0, 0)

    return results
  }
}
```
