```swift
class Solution {
  var cache = [Int: Int]()

  func tribonacci(_ n: Int) -> Int {
    if let c = cache[n] {
      return c
    }

    if n <= 0 { return 0 }
    if n < 2 { return 1 }

    let result = tribonacci(n - 3) + tribonacci(n - 2) + tribonacci(n - 1)
    cache[n] = result
    return result
  }

}
```
