/// Definition for a binary tree node.
/// public class TreeNode {
///     public var val: Int
///     public var left: TreeNode?
///     public var right: TreeNode?
///     public init() { self.val = 0; self.left = nil; self.right = nil; }
///     public init(_ val: Int) { self.val = val; self.left = nil; self.right = nil; }
///     public init(_ val: Int, _ left: TreeNode?, _ right: TreeNode?) {
///         self.val = val
///         self.left = left
///         self.right = right
///     }
/// }
class Solution {
  func leafSimilar(_ root1: TreeNode?, _ root2: TreeNode?) -> Bool {

    func dfs(_ root: TreeNode?) -> [Int] {
      guard let root else { return [] }

      if root.left == nil, root.right == nil {
        return [root.val]
      } else {
        return dfs(root.left) + dfs(root.right)
      }
    }

    let result1 = dfs(root1)
    let result2 = dfs(root2)

    return result1 == result2
  }

  func leafSimilar2(_ root1: TreeNode?, _ root2: TreeNode?) -> Bool {
    var result1 = [Int]()
    var result2 = [Int]()

    func dfs(_ root: TreeNode?, isFirstTree: Bool) {
      guard let root else { return }

      dfs(root.left, isFirstTree: isFirstTree)
      dfs(root.right, isFirstTree: isFirstTree)

      if root.left == nil, root.right == nil {
        if isFirstTree {
          result1.append(root.val)
        } else {
          result2.append(root.val)
        }
      }
    }

    dfs(root1, isFirstTree: true)
    dfs(root2, isFirstTree: false)

    return result1 == result2
  }
}
