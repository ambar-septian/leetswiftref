```swift
class Solution {
  func letterCombinations(_ digits: String) -> [String] {
    let consts: [Character: [Character]] = [
      "2": ["a", "b", "c"],
      "3": ["d", "e", "f"],
      "4": ["g", "h", "i"],
      "5": ["j", "k", "l"],
      "6": ["m", "n", "o"],
      "7": ["p", "q", "r", "s"],
      "8": ["t", "u", "v"],
      "9": ["w", "x", "y", "z"],
    ]
    var results = [String]()
    var path = [Character]()
    var digs = Array(digits)

    func backTrack(_ index: Int) {
      if path.count == digits.count {
        let string = String(path)
        results.append(string)
        return
      }

      var chars = consts[digs[index], default: []]

      for j in 0..<chars.count {
        path.append(chars[j])
        backTrack(index + 1)

        path.removeLast()
      }
    }

    backTrack(0)
    return Array(results)
  }
}
```
