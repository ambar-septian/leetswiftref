```swift
class Solution {
  func longestMonotonicSubarray(_ nums: [Int]) -> Int {
    let count = nums.count
    var currentIncrease = 1
    var currentDecrease = 1
    var maxIncrease = 1
    var maxDecrease = 1

    if count <= 1 { return count }

    for i in 1..<count {
      let curr = nums[i]
      let prev = nums[i - 1]

      if curr > prev {
        currentIncrease += 1
        maxIncrease = max(maxIncrease, currentIncrease)

        currentDecrease = 1
      } else if curr < prev {
        currentDecrease += 1
        maxDecrease = max(maxDecrease, currentDecrease)

        currentIncrease = 1
      } else {
        currentIncrease = 1
        currentDecrease = 1
      }
    }

    return max(maxDecrease, maxIncrease)
  }
}
```
