```swift
class Solution {
  func findPeakElement(_ nums: [Int]) -> Int {
    var l = 0
    let c = nums.count
    var r = c - 1

    guard c > 1 else { return 0 }

    while l <= r {
      let m = l + (r - l) / 2

      // Move from right
      if m > 0, nums[m] < nums[m - 1] {
        r = m - 1
        // Move from left
      } else if m < c - 1, nums[m] < nums[m + 1] {
        l = m + 1
      } else {
        return m
      }
    }

    return 0
  }
}
```
