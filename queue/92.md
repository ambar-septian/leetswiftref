/// Definition for singly-linked list.
/// public class ListNode {
///     public var val: Int
///     public var next: ListNode?
///     public init() { self.val = 0; self.next = nil; }
///     public init(_ val: Int) { self.val = val; self.next = nil; }
///     public init(_ val: Int, _ next: ListNode?) { self.val = val; self.next = next; }
/// }
class Solution {
  func reverseBetween(_ head: ListNode?, _ left: Int, _ right: Int) -> ListNode? {
    var head = head
    var curr = head
    var prev: ListNode?

    let target = right - left + 1

    for i in 1..<left {
      prev = curr
      curr = curr?.next
    }

    var lastNodeBeforeSublist = prev
    var lastNodeOnSublist = curr

    var prevReversed: ListNode?
    for i in 0..<target {
      let next = curr?.next
      curr?.next = prevReversed
      prevReversed = curr
      curr = next
    }

    if lastNodeBeforeSublist != nil {
      lastNodeBeforeSublist?.next = prevReversed
    } else {
      head = prevReversed
    }

    lastNodeOnSublist?.next = curr

    return head
  }
}
