```swift
class Solution {
  func pivotIndex(_ nums: [Int]) -> Int {
    let count = nums.count
    var prefixs = Array(repeating: 0, count: count)

    for i in 0..<count {
      if i == 0 {
        prefixs[0] = nums[0]
        continue
      }
      prefixs[i] = prefixs[i - 1] + nums[i]
    }

    if prefixs[0] - prefixs[count - 1] == 0 { return 0 }

    for i in 0..<count {
      if i == 0 { continue }
      if prefixs[i] == prefixs[count - 1] - prefixs[i - 1] {
        return i
      }
    }

    return -1
  }
}
```
