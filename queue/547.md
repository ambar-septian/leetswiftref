class Solution {
  class UnionSet {
    var parent = [Int]()
    var rank = [Int]()

    init(_ n: Int) {
      self.parent = (0..<n).map { $0 }
      self.rank = Array(repeating: 0, count: n)
    }

    func find(_ i: Int) -> Int {
      if parent[i] == i {
        return i
      }

      return find(parent[i])
    }

    func unite(_ a: Int, _ b: Int) -> Bool {
      let rootA = find(a)
      let rootB = find(b)

      guard rootA != rootB else { return false }

      if rank[rootA] < rank[rootB] {
        parent[rootA] = rootB
      } else if rank[rootA] > rank[rootB] {
        parent[rootB] = rootA
      } else {
        parent[rootA] = rootB
        rank[rootB] += 1
      }

      return true
    }
  }

  func findCircleNum(_ isConnected: [[Int]]) -> Int {
    /*
     1, 1, 0
     1, 1, 0,
     0, 0, 1
    */

    /*
     1, 0, 0
     0, 1, 0,
     0, 0, 1
    */
    var results = isConnected.count
    let unionSet = UnionSet(isConnected.count)

    for i in 0..<isConnected.count {
      for j in i + 1..<isConnected.count {
        if isConnected[i][j] == 1 {
          if unionSet.unite(i, j) {
            results -= 1
          }

        }
      }
    }

    return results
  }

  func findCircleNumDFS(_ isConnected: [[Int]]) -> Int {
    /*
     1, 1, 0
     1, 1, 0,
     0, 0, 1
    */

    /*
     1, 0, 0
     0, 1, 0,
     0, 0, 1
    */

    var results = 0
    var visited = Array(repeating: false, count: isConnected.count)

    func dfs(_ cityIndex: Int) {
      visited[cityIndex] = true
      for neighborIndex in 0..<isConnected[cityIndex].count {
        if !visited[neighborIndex], isConnected[cityIndex][neighborIndex] == 1 {
          dfs(neighborIndex)
        }
      }
    }

    for cityIndex in 0..<isConnected.count {
      if !visited[cityIndex] {
        dfs(cityIndex)
        results += 1
      }
    }

    return results
  }
}
