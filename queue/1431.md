```swift
class Solution {
  func kidsWithCandies(_ candies: [Int], _ extraCandies: Int) -> [Bool] {
    let max = candies.max()!
    let count = candies.count
    var result = Array(repeating: false, count: count)

    for i in 0..<count {
      if candies[i] + extraCandies >= max {
        result[i] = true
      }
    }

    return result
  }
}
```
