```swift
class Solution {
  var memo: [Int: Int] = [:]

  func climbStairs(_ n: Int) -> Int {
    if n == 0 {
      return 1
    } else if n == 1 {
      return 1
    } else {
      if let memo = memo[n] {
        return memo
      } else {
        let result = climbStairs(n - 1) + climbStairs(n - 2)
        memo[n] = result
        return result
      }
    }
  }
}
```
