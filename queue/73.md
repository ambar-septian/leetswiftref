class Solution {
  func setZeroes(_ matrix: inout [[Int]]) {
    var row = Set<(Int)>()
    var column = Set<(Int)>()

    for i in 0..<matrix.count {
      for j in 0..<matrix[i].count {
        if matrix[i][j] == 0 {
          row.insert(i)
          column.insert(j)
        }
      }
    }

    if matrix.count == 1 && (!row.isEmpty || !column.isEmpty) {
      for j in 0..<matrix[0].count {
        matrix[0][j] = 0
      }
      return
    }

    for i in 0..<matrix.count {
      for j in 0..<matrix[i].count {

        if row.contains(i) || column.contains(j) {
          matrix[i][j] = 0
        }
      }
    }
  }
}
