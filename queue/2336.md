```swift
import Collections

class SmallestInfiniteSet {
  var heap: Heap<Int>
  var uniqueSet: Set<Int>
  var nextAvailable: Int

  init() {
    heap = Heap()
    heap.reserveCapacity(1000)
    uniqueSet = Set()
    uniqueSet.reserveCapacity(1000)
    nextAvailable = 1
  }

  func popSmallest() -> Int {
    if heap.isEmpty {
      let currentAvailable = nextAvailable
      nextAvailable += 1
      return currentAvailable
    } else {
      guard let min = heap.popMin() else {
        assertionFailure("Not possible")
        return -1
      }

      uniqueSet.remove(min)
      return min
    }

  }

  func addBack(_ num: Int) {
    if num >= nextAvailable {
      return
    }

    if uniqueSet.contains(num) {
      return
    }

    uniqueSet.insert(num)
    heap.insert(num)

  }

  /*
Problem: Smallest Infinite Set (Heap + Set + Counter)
Time Complexity:
    - popSmallest: O(log M) if returning an 'added back' element, else O(1).
    - addBack: O(log M) to maintain heap and set.
    - (M is the number of elements currently 'added back').
Auxiliary Space: O(M)
    - We only store elements that have been added back.
    - The rest of the "infinite" set is represented by a single integer (nextAvailable).
*/
}

class SmallestInfiniteSet_NotEfficient {
  var heap: Heap<Int>
  var uniqueSet: Set<Int>

  init() {
    let positiveNumbers = (1...1000).map { $0 }
    heap = Heap(positiveNumbers)
    uniqueSet = Set(positiveNumbers)
  }

  func popSmallest() -> Int {
    guard let min = heap.popMin() else { return 0 }  // equal to remove
    uniqueSet.remove(min)

    return min

  }

  func addBack(_ num: Int) {
    guard !uniqueSet.contains(num) else { return }
    uniqueSet.insert(num)
    heap.insert(num)
  }
}
```
