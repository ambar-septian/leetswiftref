```swift
class Solution {
  func letterCasePermutation(_ s: String) -> [String] {
    var results = [String]()
    var path = [Character]()
    var chars = Array(s)

    func dfs(_ i: Int) {
      if path.count == chars.count {
        let string = String(path)
        results.append(string)
        return
      }

      var char = chars[i]
      path.append(char)
      dfs(i + 1)
      path.removeLast()

      guard !char.isNumber else { return }

      if char.isUppercase {
        char = Character(char.lowercased())
      } else {
        char = Character(char.uppercased())
      }

      path.append(char)
      dfs(i + 1)
      path.removeLast()
    }

    dfs(0)
    return results
  }
}
```
