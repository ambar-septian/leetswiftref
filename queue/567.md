```swift
class Solution {
  func checkInclusion(_ s1: String, _ s2: String) -> Bool {
    let chars1 = Array(s1)
    let chars2 = Array(s2)

    guard chars2.count >= chars1.count else { return false }

    var chars1Dict = [Character: Int]()
    var chars2Dict = [Character: Int]()

    for i in 0..<chars1.count {
      chars1Dict[chars1[i], default: 0] += 1
      chars2Dict[chars2[i], default: 0] += 1
    }

    if chars2Dict == chars1Dict {
      return true
    }

    for i in chars1.count..<chars2.count {
      let prev = i - chars1.count
      chars2Dict[chars2[prev], default: 0] -= 1
      if chars2Dict[chars2[prev]] == 0 {
        chars2Dict[chars2[prev]] = nil
      }
      chars2Dict[chars2[i], default: 0] += 1

      if chars2Dict == chars1Dict {
        return true
      }
    }
    return false
  }
}
```
