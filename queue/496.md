```swift
class Solution {
  func nextGreaterElement(_ nums1: [Int], _ nums2: [Int]) -> [Int] {
    var dict = [Int: Int]()
    let nums1count = nums1.count
    let nums2count = nums2.count

    for i in 0..<nums1count {
      dict[nums1[i]] = i
    }

    var stack = [Int]()
    var result = Array(repeating: -1, count: nums1count)
    for i in 0..<nums2count {
      let curr = nums2[i]
      while let last = stack.last, curr > last {
        let val = stack.removeLast()
        let index = dict[val]

        if let index {
          result[index] = curr
        }
      }

      if dict[nums2[i]] != nil {
        stack.append(nums2[i])
      }
    }

    return result
  }
}
```
