class Solution {
  func canFinish(_ numCourses: Int, _ prerequisites: [[Int]]) -> Bool {
    guard !prerequisites.isEmpty else { return true }

    var results = [Int]()
    var inDegree = Array(repeating: 0, count: numCourses)  // Node : Count
    var adjList = [Int: [Int]]()

    for pair in prerequisites {
      let course = pair[0]
      let preq = pair[1]
      adjList[preq, default: []].append(course)
      inDegree[course] += 1

    }

    var queue = [Int]()

    for (index, count) in inDegree.enumerated() {
      if count == 0 {
        queue.append(index)
      }
    }

    while !queue.isEmpty {
      let node = queue.removeFirst()
      results.append(node)

      guard let neighbors = adjList[node] else { continue }

      for neighbor in neighbors {
        inDegree[neighbor] -= 1
        if inDegree[neighbor] == 0 {
          queue.append(neighbor)
        }
      }
    }

    return results.count == numCourses

    /*
Problem: Course Schedule (Kahn's Algorithm / Topological Sort)
Time Complexity: O(V + E)
    - V: numCourses, E: prerequisites.count.
    - We build the graph in O(E) and process each node/edge in O(V + E).
    - (Assumes O(1) dequeue; Swift's removeFirst() is O(K)).
Auxiliary Space: O(V + E)
    - O(V + E) for the adjacency list.
    - O(V) for the in-degree array, queue, and results.
*/
  }

  func canFinishDict(_ numCourses: Int, _ prerequisites: [[Int]]) -> Bool {
    guard !prerequisites.isEmpty else { return true }

    var results = [Int]()
    var inDegree = [Int: Int]()  // Node : Count
    var adjList = [Int: [Int]]()

    for i in 0..<numCourses {
      inDegree[i] = 0
    }

    for pair in prerequisites {
      let course = pair[0]
      let preq = pair[1]
      adjList[preq, default: []].append(course)
      inDegree[course, default: 0] += 1

    }

    var queue = [Int]()

    for node in inDegree {
      if node.value == 0 {
        queue.append(node.key)
      }
    }

    while !queue.isEmpty {
      let node = queue.removeFirst()
      results.append(node)

      guard let neighbors = adjList[node] else { continue }

      for neighbor in neighbors {
        inDegree[neighbor, default: 0] -= 1
        if inDegree[neighbor, default: 0] == 0 {
          queue.append(neighbor)
        }
      }
    }

    return results.count == numCourses
  }
}
