```swift
class Solution {
  func plusOne(_ digits: [Int]) -> [Int] {
    var digits = digits
    var last = digits.count - 1
    while last >= 0 {
      if digits[last] != 9 {
        digits[last] = digits[last] + 1
        break
      }

      digits[last] = 0
      last -= 1

      if last < 0 {
        digits.insert(1, at: 0)
      }
    }

    return digits
  }
}
```
