```swift
class Solution {
  func productExceptSelf(_ nums: [Int]) -> [Int] {
    let count = nums.count
    var results = Array(repeating: 1, count: count)

    var left = 1
    for i in 0..<count {
      results[i] = results[i] * left
      left *= nums[i]
    }

    var right = 1
    for i in stride(from: count - 1, through: 0, by: -1) {
      results[i] = results[i] * right
      right *= nums[i]
    }

    return results
  }
}
```
