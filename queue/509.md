```swift
class Solution {
  var memo = [Int: Int]()
  func fib(_ n: Int) -> Int {
    if n == 0 { return 0 }
    if n <= 2 { return 1 }

    if let result = memo[n] {
      return result
    } else {
      let result = fib(n - 1) + fib(n - 2)
      memo[n] = result

      return result
    }
  }
}
```
