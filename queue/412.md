```swift
class Solution {
  func fizzBuzz(_ n: Int) -> [String] {
    return (1...n).map { i in
      if i % 3 == 0 && i % 5 == 0 {
        return "FizzBuzz"
      } else if i % 3 == 0 {
        return "Fizz"
      } else if i % 5 == 0 {
        return "Buzz"
      } else {
        return "\(i)"
      }
    }
  }
}
```
