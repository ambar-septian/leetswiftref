```swift
class Solution {
  func numIslands(_ grid: [[Character]]) -> Int {
    guard !grid.isEmpty else { return 0 }

    var results = 0
    var rows = grid.count
    var cols = grid[0].count
    var grid = grid
    var visited = Set<[Int]>()
    let directions: [(Int, Int)] = [(-1, 0), (1, 0), (0, 1), (0, -1)]

    func dfs(_ r: Int, _ c: Int) {
      guard r >= 0,
        c >= 0,
        r < rows,
        c < cols,
        grid[r][c] != "0", !visited.contains([r, c])
      else {
        return
      }
      // Mark visited
      grid[r][c] = "0"
      visited.insert([r, c])
      for dir in directions {
        var dirRow = r + dir.0
        var dirCol = c + dir.1
        // Iterate all directions
        dfs(dirRow, dirCol)
      }
    }

    for r in 0..<rows {
      for c in 0..<cols {
        if grid[r][c] != "0" {
          // Island found
          results += 1
          // Find all neighbor
          dfs(r, c)
        }
      }
    }

    /*
    Problem: Number of Islands (DFS Sink-the-Island)
    Time Complexity: O(M * N)
        - M: Number of rows, N: Number of columns.
        - Each cell is visited at most once.
    Auxiliary Space: O(M * N)
        - O(M * N) for the recursion stack in the worst case (all land).
        - O(M * N) for the local grid copy in Swift (triggered by modification).
    */

    return results
  }
}
```
