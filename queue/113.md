/// Definition for a binary tree node.
/// public class TreeNode {
///     public var val: Int
///     public var left: TreeNode?
///     public var right: TreeNode?
///     public init() { self.val = 0; self.left = nil; self.right = nil; }
///     public init(_ val: Int) { self.val = val; self.left = nil; self.right = nil; }
///     public init(_ val: Int, _ left: TreeNode?, _ right: TreeNode?) {
///         self.val = val
///         self.left = left
///         self.right = right
///     }
/// }
class Solution {
  func pathSum(_ root: TreeNode?, _ targetSum: Int) -> [[Int]] {
    var results = [[Int]]()
    var paths = [Int]()

    func dfs(_ root: TreeNode?, _ sum: Int) {
      guard let root else { return }

      paths.append(root.val)
      var newSum = sum + root.val

      if root.left == nil, root.right == nil, newSum == targetSum {
        results.append(paths)

      }

      dfs(root.left, newSum)
      dfs(root.right, newSum)

      paths.removeLast()
    }

    dfs(root, 0)

    return results

  }

  func pathSumNoSideEffect(_ root: TreeNode?, _ targetSum: Int) -> [[Int]] {

    func dfs(_ root: TreeNode?, _ remaining: Int) -> [[Int]] {
      guard let root else { return [] }

      let newRemaining = remaining - root.val  // 22 -5 = 17 - 8 -> 9 - 4 >..

      if root.left == nil, root.right == nil {
        return newRemaining == 0 ? [[root.val]] : []
      }

      let leftPaths = dfs(root.left, newRemaining)
      let rightPaths = dfs(root.right, newRemaining)

      return (leftPaths + rightPaths).map { [root.val] + $0 }
    }

    return dfs(root, targetSum)
  }

  func pathSum2(_ root: TreeNode?, _ targetSum: Int) -> [[Int]] {
    var results = [[Int]]()

    func dfs(_ root: TreeNode?, sum: Int, nums: [Int]) {
      guard let root else {
        return
      }

      var newSum = sum + root.val
      var newNums = nums
      newNums.append(root.val)

      if root.left == nil, root.right == nil, newSum == targetSum {
        results.append(newNums)
        return
      }

      dfs(root.left, sum: newSum, nums: newNums)
      dfs(root.right, sum: newSum, nums: newNums)
    }

    dfs(root, sum: 0, nums: [])
    return results
  }
}
