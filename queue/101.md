/// Definition for a binary tree node.
/// public class TreeNode {
///     public var val: Int
///     public var left: TreeNode?
///     public var right: TreeNode?
///     public init() { self.val = 0; self.left = nil; self.right = nil; }
///     public init(_ val: Int) { self.val = val; self.left = nil; self.right = nil; }
///     public init(_ val: Int, _ left: TreeNode?, _ right: TreeNode?) {
///         self.val = val
///         self.left = left
///         self.right = right
///     }
/// }
class Solution {
  func isSymmetric(_ root: TreeNode?) -> Bool {
    var queue = [root?.left, root?.right]

    while !queue.isEmpty {
      let l1 = queue.removeFirst()
      let l2 = queue.removeFirst()

      if l1 == nil && l2 == nil { continue }

      let l1Left = l1?.left
      let l2Right = l2?.right
      let l1Right = l1?.right
      let l2Left = l2?.left

      if l1?.val != l2?.val {
        return false
      }

      // if l1Left?.val != l2Right?.val && l1Right?.val != l2Left?.val  {
      //     return false
      // }

      queue.append(contentsOf: [l1Left, l2Right, l1Right, l2Left])

    }


    return true
  }
}
