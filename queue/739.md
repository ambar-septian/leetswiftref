```swift
class Solution {
  func dailyTemperatures(_ temperatures: [Int]) -> [Int] {
    var stack = [Int]()
    var results = Array(repeating: 0, count: temperatures.count)

    for i in 0..<temperatures.count {
      while !stack.isEmpty, let topIndex = stack.last, temperatures[i] > temperatures[topIndex] {
        stack.removeLast()
        results[topIndex] = i - topIndex
      }

      stack.append(i)
    }

    return results

  }
}
```
