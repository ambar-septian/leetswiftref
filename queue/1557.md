```swift
class Solution {
  func findSmallestSetOfVertices(_ n: Int, _ edges: [[Int]]) -> [Int] {
    var verticies = Set<Int>()
    var list = [Int]()
    for edge in edges {
      verticies.insert(edge[0])
      list.append(edge[1])
    }

    for l in list {
      verticies.remove(l)
    }
    return Array(verticies)
  }
}
```
