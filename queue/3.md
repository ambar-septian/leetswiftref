```swift
class Solution {
  func lengthOfLongestSubstring(_ s: String) -> Int {
    let chars = Array(s)
    var cache = Set<Character>()
    var result = 0
    var left = 0
    var right = 0

    while right < chars.count {
      if !cache.contains(chars[right]) {
        cache.insert(chars[right])
        result = max(result, right - left + 1)
        right += 1
      } else {
        cache.remove(chars[left])
        left += 1
      }
    }

    return result
  }
}
```
