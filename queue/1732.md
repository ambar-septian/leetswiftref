```swift
class Solution {
  func largestAltitude(_ gain: [Int]) -> Int {
    var sum = 0
    var maxAlt = 0

    for i in 0..<gain.count {
      sum += gain[i]
      maxAlt = max(sum, maxAlt)
    }

    return maxAlt

  }
}
```
