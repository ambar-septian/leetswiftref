class Solution {
  func isValidSudoku(_ board: [[Character]]) -> Bool {
    var rows = Array(repeating: Set<Character>(), count: 9)
    var cols = Array(repeating: Set<Character>(), count: 9)
    var boxes = Array(repeating: Set<Character>(), count: 9)
    let rowsCount = board.count
    let columnsCount = board[0].count

    for i in 0..<rowsCount {
      for j in 0..<columnsCount {
        let item = board[i][j]
        guard item != "." else { continue }

        if rows[i].contains(item) {
          return false
        }
        rows[i].insert(item)

        if cols[j].contains(item) {
          return false
        }
        cols[j].insert(item)

        let boxIndex = (i / 3) * 3 + (j / 3)
        if boxes[boxIndex].contains(item) {
          return false
        }
        boxes[boxIndex].insert(item)
      }
    }

    return true
  }

  func isValidSudoku2(_ board: [[Character]]) -> Bool {
    var rowDict = [Int: Set<Character>]()
    var colDict = [Int: Set<Character>]()
    var boxDict = [Int: Set<Character>]()

    for i in 0..<board.count {
      for j in 0..<board[i].count {
        let item = board[i][j]
        guard item != "." else { continue }
        var rowSet: Set<Character>
        var colSet: Set<Character>
        var boxSet: Set<Character>

        if var hashSet = rowDict[i] {
          rowSet = hashSet
          if rowSet.contains(item) {
            return false
          }
        } else {
          rowSet = Set<Character>()
        }
        rowSet.insert(item)
        rowDict[i] = rowSet

        if var hashSet = colDict[j] {
          colSet = hashSet
          if colSet.contains(item) {
            return false
          }
        } else {
          colSet = Set<Character>()
        }
        colSet.insert(item)
        colDict[j] = colSet

        let boxIndex = (i / 3) * 3 + (j / 3)
        if var hashSet = boxDict[boxIndex] {
          boxSet = hashSet
          if boxSet.contains(item) {
            return false
          }
        } else {
          boxSet = Set<Character>()
        }
        boxSet.insert(item)
        boxDict[boxIndex] = boxSet
      }
    }

    return true
  }
}
