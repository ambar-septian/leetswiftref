class Solution {
  func orangesRotting(_ grid: [[Int]]) -> Int {
    guard !grid.isEmpty else { return 0 }

    let rows = grid.count
    let cols = grid[0].count
    let directions = [(0, 1), (1, 0), (0, -1), (-1, 0)]

    var grid = grid
    var queue = [Int]()
    var freshCount = 0
    var results = 0

    for r in 0..<rows {
      for c in 0..<cols {
        if grid[r][c] == 2 {
          let item = r * cols + c
          queue.append(item)
        } else if grid[r][c] == 1 {
          freshCount += 1
        }
      }
    }

    while !queue.isEmpty {
      let count = queue.count
      var rootenFound = false
      for _ in 0..<count {
        let evidict = queue.removeFirst()
        let rIndex = evidict / cols
        let cIndex = evidict % cols

        for dir in directions {
          let rDir = rIndex + dir.0
          let cDir = cIndex + dir.1
          if rDir < 0 || cDir < 0 || rDir >= rows || cDir >= cols {
            continue
          }

          if grid[rDir][cDir] == 1 {
            grid[rDir][cDir] = 2
            queue.append(rDir * cols + cDir)

            rootenFound = true
            freshCount -= 1
          }

        }
      }

      if rootenFound {
        results += 1
      }
    }

    return freshCount == 0 ? results : -1

  }

  /*
Problem: Rotting Oranges (Multi-source BFS)
Time Complexity: O(M * N)
    - M: Number of rows, N: Number of columns.
    - Each cell is visited at most once during the initial scan and BFS.
Auxiliary Space: O(M * N)
    - O(M * N) for the BFS queue in the worst case.
    - O(M * N) for the local grid copy in Swift.
*/
}
