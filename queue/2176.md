```swift
class Solution {
  func countPairs(_ nums: [Int], _ k: Int) -> Int {
    var dict = [Int: [Int]]()
    var result = 0

    for i in 0..<nums.count {
      let num = nums[i]

      if var existing = dict[num] {

        for j in 0..<existing.count {
          let divisible = (i * existing[j]) % k == 0
          let value = divisible ? 1 : 0
          result += value
        }

        existing.append(i)
        dict[num] = existing
      } else {
        dict[num] = [i]
      }
    }

    return result
  }
}
```
