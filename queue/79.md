class Solution {
  func exist(_ board: [[Character]], _ word: String) -> Bool {
    var chars = Array(word)
    var board = board
    let row = board.count
    let col = board[0].count
    let count = chars.count

    guard row * col >= count else { return false }

    let firstCharPositions = firstCharInGrid(board, word)
    guard !firstCharPositions.isEmpty else { return false }

    func dfs(_ r: Int, _ c: Int, _ charIndex: Int) -> Bool {
      if charIndex == count {
        return true
      }

      if r < 0 || c < 0 || r >= row || c >= col || board[r][c] != chars[charIndex] {
        return false
      }

      let temp = board[r][c]
      board[r][c] = "#"  // replace the board instead of using set

      let nextIndex = charIndex + 1
      let result =
        dfs(r + 1, c, nextIndex) || dfs(r - 1, c, nextIndex) || dfs(r, c + 1, nextIndex)
        || dfs(r, c - 1, nextIndex)

      board[r][c] = temp

      return result
    }

    for firstChar in firstCharPositions {
      if dfs(firstChar.0, firstChar.1, 0) {
        return true
      }
    }

    return false

    /*
Problem: Word Search (with Frequency Pruning)
Time Complexity: O(N * 3^L)
    - N: Total number of cells (Rows * Columns).
    - L: Length of the word.
    - Pruning check takes O(N + L), which is additive and doesn't change the Big O upper bound.
    - Backtracking remains the bottleneck at O(N * 3^L).
Auxiliary Space: O(N + L)
    - O(N) for the local board copy (triggered by Copy-on-Write modification).
    - O(L) for the recursion stack and frequency map.
*/
  }

  func firstCharInGrid(_ board: [[Character]], _ word: String) -> [(Int, Int)] {
    var freq = [Character: Int]()
    var firstCharPositions = [(Int, Int)]()

    for char in word {
      freq[char, default: 0] += 1
    }
    for i in 0..<board.count {
      for j in 0..<board[0].count {
        let char = board[i][j]
        if char == word.first {
          firstCharPositions.append((i, j))
        }

        if freq[char] != nil {
          freq[char]! -= 1
          if freq[char]! <= 0 { freq[char] = nil }
        }
      }
    }

    if freq.isEmpty {
      return firstCharPositions
    } else {
      return []
    }
  }
}
