```swift
class Solution {
  func removeDuplicates(_ nums: inout [Int]) -> Int {
    guard !nums.isEmpty else { return 0 }

    var slow = 0
    var fast = 1
    var result = 1

    while fast < nums.count {
      if nums[slow] == nums[fast] {
        fast += 1

      } else {
        slow += 1
        nums[slow] = nums[fast]

        result += 1
      }
    }
    
    return result
  }
}
```
