```swift
class Solution {
  func reverseVowels(_ s: String) -> String {
    var chars = s.map { $0 }
    var left = 0
    var right = chars.count - 1
    var vowels: Set<Character> = ["a", "A", "i", "I", "u", "U", "e", "E", "o", "O"]

    while left < right {
      let l = chars[left]
      let r = chars[right]

      if vowels.contains(l), vowels.contains(r) {
        chars[left] = r
        chars[right] = l
        left += 1
        right -= 1
      } else if vowels.contains(l) {
        right -= 1
      } else if vowels.contains(r) {
        left += 1
      } else {
        left += 1
        right -= 1
      }
    }

    return String(chars)
  }
}
```
