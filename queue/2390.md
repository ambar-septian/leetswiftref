```swift
class Solution {
  func removeStars(_ s: String) -> String {
    var result = [Character]()
    result.reserveCapacity(s.count)

    for char in s {
      if char == "*" {
        result.removeLast()
      } else {
        result.append(char)
      }
    }

    return String(result)
  }
}
```
