```swift
class Solution {
  class UnionFind {
    var parent = [Int]()
    var rank = [Int]()

    init(_ n: Int) {
      parent = (0..<n).map { $0 }
      rank = Array(repeating: 0, count: n)
    }

    func find(_ i: Int) -> Int {
      if parent[i] == i {
        return i
      }

      parent[i] = find(parent[i])
      return parent[i]
    }

    func unite(_ a: Int, _ b: Int) -> Bool {
      let rootA = find(a)
      let rootB = find(b)

      guard rootA != rootB else { return false }

      if rank[rootA] < rank[rootB] {
        parent[rootA] = rootB
      } else if rank[rootA] > rank[rootB] {
        parent[rootB] = rootA
      } else {
        rank[rootB] += 1
        parent[rootA] = rootB
      }

      return true
    }
  }

  func accountsMerge(_ accounts: [[String]]) -> [[String]] {
    var emailToAccIndex = [String: Int]()
    let unionFind = UnionFind(accounts.count)

    for (i, account) in accounts.enumerated() {
      for j in (1..<account.count) {  // Skip zero due the first item is name
        let email = account[j]
        if let existingIndex = emailToAccIndex[email] {
          unionFind.unite(i, existingIndex)
        } else {
          emailToAccIndex[email] = i
        }
      }
    }

    var emailGroup = [Int: [String]]()

    for (email, i) in emailToAccIndex {
      let parent = unionFind.find(i)
      emailGroup[parent, default: []].append(email)
    }

    var results = [[String]]()
    for (i, emails) in emailGroup {
      let name = accounts[i][0]
      var result = [name] + emails.sorted()
      results.append(result)
    }

    return results
  }
}
```
