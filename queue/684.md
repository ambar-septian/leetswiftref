```swift
class Solution {
  class UnionSet {
    var parent = [Int]()
    var rank = [Int]()

    init(_ n: Int) {
      self.parent = (0..<n).map { $0 }
      self.rank = Array(repeating: 0, count: n)
    }

    func find(_ i: Int) -> Int {
      if parent[i] == i {
        return i
      }

      parent[i] = find(parent[i])
      return parent[i]
    }

    func unite(_ a: Int, _ b: Int) -> Bool {
      let rootA = find(a)
      let rootB = find(b)
      guard rootA != rootB else { return false }

      if rank[rootA] < rank[rootB] {
        parent[rootA] = rootB
      } else if rank[rootA] > rank[rootB] {
        parent[rootB] = rootA
      } else {
        parent[rootA] = rootB
        rank[rootB] += 1
      }

      return true
    }
  }

  func findRedundantConnection(_ edges: [[Int]]) -> [Int] {
    let unionSet = UnionSet(edges.count)
    var result = [Int]()

    for edge in edges {
      // If it has same parentIndex than return true, since it doest need to unite
      if !unionSet.unite(edge[0] - 1, edge[1] - 1) {
        result = [edge[0], edge[1]]
      }
    }

    return result

  }

  /*
Problem: Redundant Connection (Union-Find)
Time Complexity: O(N * α(N))
    - N: Number of nodes/edges.
    - α(N): Inverse Ackermann function (effectively constant).
    - Achieved via Path Compression and Union by Rank.
Auxiliary Space: O(N)
    - O(N) to store 'parent' and 'rank' arrays.
    - Path compression keeps the recursion stack extremely shallow.
*/
}
```
