class Solution {
  func validPath(_ n: Int, _ edges: [[Int]], _ source: Int, _ destination: Int) -> Bool {
    var visited = Array(repeating: false, count: n)
    var list = Array(repeating: [Int](), count: n)

    guard source != destination else { return true }

    for edge in edges {
      list[edge[0]].append(edge[1])
      list[edge[1]].append(edge[0])
    }

    func dfs(_ curr: Int) -> Bool {
      let edge = list[curr]

      for neighbhor in edge {
        if neighbhor == destination {
          return true
        } else if !visited[neighbhor] {
          visited[neighbhor] = true
          if dfs(neighbhor) {
            return true
          }
        }
      }

      return false
    }

    return dfs(source)

    /*
Problem: Find if Path Exists in Graph (DFS)
Time Complexity: O(V + E)
    - V: Number of nodes (n).
    - E: Number of edges.
    - We build the adjacency list in O(E) and traverse the graph in O(V + E).
Auxiliary Space: O(V + E)
    - O(V + E) to store the adjacency list.
    - O(V) for the 'visited' array and the recursion stack depth.
*/
  }

  func validPathHash(_ n: Int, _ edges: [[Int]], _ source: Int, _ destination: Int) -> Bool {
    var visited = Set<Int>()
    var list = [Int: [Int]]()

    guard source != destination else { return true }

    for edge in edges {
      list[edge[0], default: []].append(edge[1])
      list[edge[1], default: []].append(edge[0])
    }

    func dfs(_ curr: Int) -> Bool {
      guard let edge = list[curr] else {
        return false
      }

      for neighbhor in edge {
        if neighbhor == destination {
          return true
        } else if !visited.contains(neighbhor) {
          visited.insert(neighbhor)
          if dfs(neighbhor) {
            return true
          }
        }
      }

      return false
    }

    return dfs(source)
  }
}
