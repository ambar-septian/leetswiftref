```swift
class Solution {
  func increasingTriplet(_ nums: [Int]) -> Bool {
    var i = 0
    var first = Int.max
    var second = Int.max

    while i < nums.count {
      if nums[i] > first && nums[i] < second {
        second = nums[i]
      } else if nums[i] < first {
        first = nums[i]
      } else if nums[i] > first && nums[i] > second {
        return true
      }
      i += 1
    }

    return false
  }
}
```
