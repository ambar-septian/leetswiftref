```swift
class ListNode {
  var val: Int?
  var next: ListNode?

  init(_ val: Int?) {
    self.val = val
    self.next = nil
  }
}

class MyHashSet {
  var array: [ListNode?]

  init() {
    array = (0..<10_000).map { _ in nil }
  }

  func add(_ key: Int) {
    let hashKey = hash(key)
    if let listNode = array[hashKey] {
      var current: ListNode? = listNode

      while current?.next != nil {
        if current?.next?.val == key { return }
        current = current?.next
      }
      current?.next = ListNode(key)

    } else {
      let listNode = ListNode(nil)  // dummy
      listNode.next = ListNode(key)
      array[hashKey] = listNode
    }
  }

  func remove(_ key: Int) {
    let hashKey = hash(key)
    guard let listNode = array[hashKey] else {
      return
    }
    var current: ListNode? = listNode
    while current != nil {
      if current?.next?.val == key {
        current?.next = current?.next?.next
        break
      }
      current = current?.next
    }
    array[hashKey] = listNode
  }

  func contains(_ key: Int) -> Bool {
    let hashKey = hash(key)
    guard let listNode = array[hashKey] else {
      return false
    }

    var current: ListNode? = listNode
    while current != nil {
      if current?.val == key { return true }
      current = current?.next
    }

    return false
  }

  private func hash(_ key: Int) -> Int {
    return key % 10_000
  }
}

/**
 * Your MyHashSet object will be instantiated and called as such:
 * let obj = MyHashSet()
 * obj.add(key)
 * obj.remove(key)
 * let ret_3: Bool = obj.contains(key)
 */
```
