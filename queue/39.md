```swift
class Solution {
  func combinationSum(_ candidates: [Int], _ target: Int) -> [[Int]] {
    var results = [[Int]]()
    var paths = [Int]()
    var sums = 0

    func dfs(_ startIndex: Int) {
      if sums == target {
        results.append(paths)
        return
      } else if sums > target {
        return
      } else {
        for i in startIndex..<candidates.count {
          let c = candidates[i]
          paths.append(c)

          sums += c
          dfs(i)

          paths.removeLast()
          sums -= c
        }
      }
    }

    dfs(0)
    return results
    /*
            Problem: Combination Sum (Reuse elements allowed)
    Time Complexity: O(N^(T/M) * (T/M))
        - N: Number of candidates.
        - T: Target value.
        - M: Minimal value in candidates.
        - The T/M factor comes from the copy operation when a valid sum is found.
    Auxiliary Space: O(T/M)
        - Max depth of recursion stack is T divided by the smallest candidate.
        - 'paths' array size is also proportional to T/M.
       */
  }
}
```
