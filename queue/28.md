```swift
class Solution {
  func strStr(_ haystack: String, _ needle: String) -> Int {
    let h = haystack.map { String($0) }
    let n = needle.map { String($0) }
    guard h.count > 0, n.count > 0 else { return -1 }

    var left = 0
    var right = 0
    var count = 0

    while left < h.count {
      guard right + left < h.count, right < n.count else { return -1 }
      if h[right + left] == n[right] {
        right += 1
        count += 1

        if count == n.count {
          return left
        }
      } else {
        count = 0
        left += 1
        right = 0
      }
    }

    return -1
  }
}
```
