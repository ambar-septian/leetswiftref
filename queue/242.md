```swift
class Solution {
  func isAnagram(_ s: String, _ t: String) -> Bool {
    return isAnagramWithAscii(s, t)

    var sdict = [Character: Int]()
    var tdict = [Character: Int]()
    let schars = Array(s)
    let tchars = Array(t)
    let maxChars = max(schars.count, tchars.count)

    for i in 0..<maxChars {
      if schars.count > i {
        let schar = schars[i]
        if var curr = sdict[schar] {
          curr += 1
          sdict[schar] = curr
        } else {
          sdict[schar] = 1
        }
      }

      if tchars.count > i {
        let tchar = tchars[i]
        if var curr = tdict[tchar] {
          curr += 1
          tdict[tchar] = curr
        } else {
          tdict[tchar] = 1
        }
      }
    }

    return sdict.count == tdict.count

  }

  func isAnagramWithAscii(_ s: String, _ t: String) -> Bool {
    guard s.count == t.count else { return false }

    var counts = Array(repeating: 0, count: 26)
    for (schar, tchar) in zip(s, t) {
      counts[Int(schar.asciiValue! - 97)] += 1
      counts[Int(tchar.asciiValue! - 97)] -= 1
    }

    return counts.allSatisfy { $0 == 0 }
  }
}
```
