/// Definition for a binary tree node.
/// public class TreeNode {
///     public var val: Int
///     public var left: TreeNode?
///     public var right: TreeNode?
///     public init() { self.val = 0; self.left = nil; self.right = nil; }
///     public init(_ val: Int) { self.val = val; self.left = nil; self.right = nil; }
///     public init(_ val: Int, _ left: TreeNode?, _ right: TreeNode?) {
///         self.val = val
///         self.left = left
///         self.right = right
///     }
/// }
class Solution {
  func isValidBST(_ root: TreeNode?) -> Bool {

    func isValid(_ root: TreeNode?, left: Int, right: Int) -> Bool {
      guard let root else { return true }

      guard root.val > left, root.val < right else { return false }

      return isValid(root.left, left: left, right: root.val)
        && isValid(root.right, left: root.val, right: right)
    }

    return isValid(root, left: Int.min, right: Int.max)
  }
}
