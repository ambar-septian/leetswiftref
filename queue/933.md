```swift
class RecentCounter {
  let const = 3000
  var pings = [Int]()
  var currentIndex = 0

  init() {

  }

  func ping(_ t: Int) -> Int {
    if currentIndex > 1000 {
      pings.removeFirst(currentIndex)
      currentIndex = 0
    }

    while currentIndex < pings.count, t - pings[currentIndex] > const {
      currentIndex += 1
    }

    pings.append(t)
    return pings.count - currentIndex
  }
}

/**
 * Your RecentCounter object will be instantiated and called as such:
 * let obj = RecentCounter()
 * let ret_1: Int = obj.ping(t)
 */
```
