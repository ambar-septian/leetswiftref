```swift
import Foundation

class Solution {
  func findWordsContaining(_ words: [String], _ x: Character) -> [Int] {
    var i = 0
    var j = 0
    var results = [Int]()
    var curr: [Character]?

    while i < words.count {
      if curr == nil {
        curr = Array(words[i])
      }

      if j >= curr!.count {
        j = 0
        i += 1
        curr = nil
        continue
      }

      if curr![j] == x {
        results.append(i)
        j = 0
        i += 1
        curr = nil
      } else {
        j += 1
      }
    }

    return results
  }
}
```
