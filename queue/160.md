```swift
/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     public var val: Int
 *     public var next: ListNode?
 *     public init(_ val: Int) {
 *         self.val = val
 *         self.next = nil
 *     }
 * }
 */

class Solution {
  func getIntersectionNode(_ headA: ListNode?, _ headB: ListNode?) -> ListNode? {
    var hasLoopToB = false
    var hasLoopToA = false
    var tailA = headA
    var tailB = headB

    while tailA != nil || tailB != nil {
      if tailA === tailB {
        return tailA
      }

      tailA = tailA?.next
      tailB = tailB?.next

      if tailA == nil, !hasLoopToB {
        tailA = headB
        hasLoopToB = true
      }

      if tailB == nil, !hasLoopToA {
        tailB = headA
        hasLoopToA = true
      }
    }

    return nil
  }
}
```
