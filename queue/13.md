```swift
class Solution {
  func romanToInt(_ s: String) -> Int {
    var result = 0
    var temp = 0
    let count = s.count

    for (index, char) in s.enumerated() {
      let number: Int
      switch char {
      case "I":
        number = 1
      case "V":
        number = 5
      case "X":
        number = 10
      case "L":
        number = 50
      case "C":
        number = 100
      case "D":
        number = 500
      case "M":
        number = 1000
      default:
        number = 0
      }

      if number > temp, temp != 0 {
        result += (number - (temp * 2))
      } else {
        result += number
      }

      temp = number
    }

    return result
  }
}
```
