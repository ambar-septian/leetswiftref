```swift
class Solution {
  func mergeAlternately(_ word1: String, _ word2: String) -> String {
    let chars1 = Array(word1)
    let chars2 = Array(word2)
    let chars1Count = chars1.count
    let chars2Count = chars2.count
    let count = max(chars1Count, chars2Count)
    var result = Array(repeating: Character(" "), count: chars1Count + chars2Count)
    var j = 0
    for i in 0..<count {
      if i < chars1Count {
        result[j] = chars1[i]
        j += 1
      }
      if i < chars2Count {
        result[j] = chars2[i]
        j += 1
      }
    }

    return String(result)
  }
}
```
