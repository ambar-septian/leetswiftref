class Solution {
  let root = Node()

  func suggestedProducts(_ products: [String], _ searchWord: String) -> [[String]] {
    var sortedProducts = products.sorted()

    for product in sortedProducts {
      var current = root

      for char in product {
        if current.children[char] == nil {
          current.children[char] = Node()
        }

        current = current.children[char]!
        current.appendResult(product)
      }
    }

    var results = [[String]]()
    var curr: Node? = root
    for char in searchWord {
      let products = curr?.children[char]?.results ?? []
      results.append(products)

      curr = curr?.children[char]
    }

    return results
  }
}

class Node {
  var children = [Character: Node]()
  private(set) var results = [String]()

  func appendResult(_ string: String) {
    guard results.count < 3 else { return }
    results.append(string)
  }

  func debugPrint(prefix: String = "", depth: Int = 0) {
    let indent = String(repeating: "  ", count: depth)

    for key in children.keys.sorted() {
      let results = children[key]?.results.joined(separator: ", ") ?? ""
      print("\(indent)\(key) -- \(results)")
      children[key]?.debugPrint(prefix: prefix + String(key), depth: depth + 1)
    }
  }

}
