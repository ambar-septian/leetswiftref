```swift
class Solution {
  func partition(_ s: String) -> [[String]] {
    var results = [[String]]()
    var part = [String]()
    var chars = Array(s)
    var cache = [[Int]: String]()

    func dfs(_ i: Int) {
      if i >= chars.count {
        results.append(part)
        return
      }

      for j in i..<chars.count {
        // Optimisation by using cache
        if let exist = cache[[i, j]] {
          part.append(exist)
          dfs(j + 1)
          part.removeLast()

        } else if isPalindrome(i, j, chars) {
          let newString = String(chars[i...j])
          cache[[i, j]] = newString

          part.append(newString)
          dfs(j + 1)
          part.removeLast()
        }
      }
    }

    dfs(0)

    return results
  }

  func isPalindrome(_ l: Int, _ r: Int, _ s: [Character]) -> Bool {
    var l = l
    var r = r
    while l <= r {
      if s[l] != s[r] {
        return false
      }
      l += 1
      r -= 1
    }

    return true
  }
}
```
