```swift
class Solution {
  func numberOfSteps(_ num: Int) -> Int {
    var result = 0
    var current = num

    while current != 0 {
      if current % 2 == 0 {
        current = current / 2
      } else {
        current -= 1
      }

      result += 1
    }

    return result
  }
}
```
