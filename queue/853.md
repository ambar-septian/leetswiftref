class Solution {
  func carFleet(_ target: Int, _ position: [Int], _ speed: [Int]) -> Int {
    // create array of tuple position and speet
    // sort the array
    // loop from the latest to zero
    // if the two last item,,,

    guard position.count > 1 else { return 1 }

    let cars = Array(zip(position, speed)).sorted(by: { $0.0 > $1.0 })
    var stack = [Float]()

    for car in cars {
      let cal = (Float(target) - Float(car.0))
      let speed = Float(car.1)
      stack.append(cal / speed)
      while stack.count >= 2, stack[stack.count - 1] <= stack[stack.count - 2] {
        stack.removeLast()
      }
    }
    return stack.count
  }
}
