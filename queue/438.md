class Solution {
  func findAnagrams(_ s: String, _ p: String) -> [Int] {
    let sArray = Array(s)
    let pArray = Array(p)
    let sCount = sArray.count
    let pCount = pArray.count

    guard sCount >= pCount else { return [] }

    var result = [Int]()
    var countP = Array(repeating: 0, count: 26)
    var countS = countP

    func idx(_ c: Character) -> Int {
      Int(c.asciiValue! - Character("a").asciiValue!)
    }

    for i in 0..<p.count {
      countP[idx(pArray[i])] += 1
      countS[idx(sArray[i])] += 1
    }

    if countP == countS {
      result.append(0)
    }

    for i in pCount..<sCount {
      countS[idx(sArray[i])] += 1
      countS[idx(sArray[i - pCount])] -= 1

      if countS == countP {
        result.append(i - pCount + 1)
      }
    }

    return result
  }

  func findAnagramsNotOptimal(_ s: String, _ p: String) -> [Int] {
    guard s.count >= p.count else {
      return []
    }
    let pChars = p.map { $0 }
    let sChars = s.map { $0 }

    var pCount = [Character: Int]()
    var sCount = [Character: Int]()

    for i in 0..<p.count {
      if let pItem = pCount[pChars[i]] {
        pCount[pChars[i]] = pItem + 1
      } else {
        pCount[pChars[i]] = 1
      }

      if let sItem = sCount[sChars[i]] {
        sCount[sChars[i]] = sItem + 1
      } else {
        sCount[sChars[i]] = 1
      }
    }

    var result: [Int]
    if sCount == pCount {
      result = [0]
    } else {
      result = []
    }

    var l = 0
    for r in p.count..<s.count {
      if let sItem = sCount[sChars[r]] {
        sCount[sChars[r]] = sItem + 1
      } else {
        sCount[sChars[r]] = 1
      }

      if let sItem = sCount[sChars[l]] {
        sCount[sChars[l]] = sItem - 1
      }

      if sCount[sChars[l]] == 0 {
        sCount[sChars[l]] = nil
      }

      l += 1
      if sCount == pCount {
        result.append(l)
      }
    }

    return result
  }
}
