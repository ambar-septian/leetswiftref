```swift
class Solution {
  func groupAnagrams(_ strs: [String]) -> [[String]] {

    var dict = [String: [String]]()

    for str in strs {
      let key = String(str.sorted())  // O(k log k)
      dict[key, default: []].append(str)
    }

    return Array(dict.values)
  }

  func isValidAnagram(_ s: String, _ t: String) -> Bool {
    guard s.count == t.count else { return false }

    var counts = Array(repeating: 0, count: 26)
    for (schar, tchar) in zip(s, t) {
      counts[Int(schar.asciiValue! - 97)] += 1
      counts[Int(tchar.asciiValue! - 97)] -= 1
    }

    return counts.allSatisfy { $0 == 0 }
  }
}
```
