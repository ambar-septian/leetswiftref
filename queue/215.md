```swift
class Solution {
  struct MaxInt: Comparable {
    let value: Int

    static func < (lhs: Self, rhs: Self) -> Bool {
      lhs.value > rhs.value
    }
  }

  func findKthLargest(_ nums: [Int], _ k: Int) -> Int {
    let array = nums.map { MaxInt(value: $0) }
    var heap = Heap(nums)

    var k = k
    var result = -1
    while k > 0 {
      guard let max = heap.popMax() else {
        continue
      }
      result = max
      k -= 1
    }

    return result

    /*
Problem: Kth Largest Element in an Array (Max-Heap)
Time Complexity: O(N + k log N)
    - N: Total number of elements.
    - Building the heap takes O(N).
    - Popping the max k times takes O(k log N).
Auxiliary Space: O(N)
    - O(N) to store the heap elements.
    - O(N) for the initial mapping/conversion of values.
*/
  }
}
```
