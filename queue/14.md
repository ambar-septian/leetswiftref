```swift
class Solution {
  func longestCommonPrefix(_ strs: [String]) -> String {
    if strs.isEmpty {
      return ""
    }

    var results = [String]()
    outerLoop: for (firstCharIndex, char) in strs.first!.enumerated() {
      // firstCharIndex -> f, l, o
      for stringIndex in 1..<strs.count {

        let nextWord = strs[stringIndex]
        guard firstCharIndex < nextWord.count else { break outerLoop }
        let nextWordIndex = nextWord.index(nextWord.startIndex, offsetBy: firstCharIndex)
        if nextWord[nextWordIndex] != char {
          return results.joined()
        }
      }

      results.append(String(char))
    }

    return results.joined()
  }
}
```
