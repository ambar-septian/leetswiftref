```swift
class Solution {
  func decodeString(_ s: String) -> String {
    var stack = [String]()

    func index(_ index: Int) -> Character {
      let strIndex = s.index(s.startIndex, offsetBy: index)
      return s[strIndex]
    }

    for i in 0..<s.count {
      let str = index(i)
      if str != "]" {
        stack.append(String(str))
      } else {
        var substr = ""
        while let top = stack.last, top != "[" {
          let lastItem = stack.removeLast()
          substr = lastItem + substr
        }

        stack.removeLast()  // Pop the "["

        var k = ""
        while let top = stack.last, let isNumber = Int(top) {
          let number = stack.removeLast()
          k = number + k
        }

        stack.append(
          String(repeating: substr, count: Int(k) ?? 0)
        )
      }
    }

    return stack.joined(separator: "")
  }
}
```
