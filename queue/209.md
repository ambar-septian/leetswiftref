```swift
class Solution {
  func minSubArrayLen(_ target: Int, _ nums: [Int]) -> Int {
    var result = Int.max
    var i = 0
    var sum = 0

    for j in 0..<nums.count {
      sum += nums[j]

      while sum >= target {
        sum -= nums[i]
        result = min(result, j - i + 1)
        i += 1
      }
    }

    return result == Int.max ? 0 : result
  }
}
```
