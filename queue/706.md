class ListNode {
  var val: Int?
  var key: Int?
  var next: ListNode?

  init(_ key: Int?, _ val: Int?) {
    self.val = val
    self.key = key
    self.next = nil
  }

  var description: String {
    return "current key: \(key), \(val), next key: \(next?.key), \(next?.val)"
  }
}

class MyHashMap {
  var array: [ListNode?]

  init() {
    array = (0..<10_000).map { _ in nil }
  }

  func put(_ key: Int, _ value: Int) {
    let hashKey = hash(key)
    if let listNode = array[hashKey] {
      var current: ListNode? = listNode

      while current?.next != nil {
        // update
        if current?.next?.key == key {
          current?.next?.val = value
          return
        }
        current = current?.next
      }
      current?.next = ListNode(key, value)

    } else {
      let listNode = ListNode(nil, nil)  // dummy
      listNode.next = ListNode(key, value)
      array[hashKey] = listNode
    }
  }

  func get(_ key: Int) -> Int {
    let hashKey = hash(key)
    guard let listNode = array[hashKey] else {
      return -1
    }

    var current: ListNode? = listNode
    while current != nil {
      if current?.key == key {
        return current?.val ?? -1
      }
      current = current?.next
    }

    return -1
  }

  func remove(_ key: Int) {
    let hashKey = hash(key)
    guard let listNode = array[hashKey] else {
      return
    }
    var current: ListNode? = listNode

    while current != nil {
      if current?.next?.key == key {
        current?.next = current?.next?.next
        break
      }
      current = current?.next
    }

    array[hashKey] = listNode
  }

  private func hash(_ key: Int) -> Int {
    return key % 10_000
  }
}
