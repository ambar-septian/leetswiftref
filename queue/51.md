class Solution {
  func solveNQueens(_ n: Int) -> [[String]] {
    var results = [[String]]()
    var board = Array(repeating: Array(repeating: ".", count: n), count: n)
    var colSet = Set<Int>()
    var posDigSet = Set<Int>()  // r + c
    var negDigSet = Set<Int>()  // r - c

    func dfs(_ r: Int) {
      if r == n {
        let result = board.map {
          $0.joined(separator: "")
        }
        results.append(result)
        return
      }

      for c in 0..<n {
        if colSet.contains(c) || posDigSet.contains(r + c) || negDigSet.contains(r - c) {
          continue
        }

        board[r][c] = "Q"
        colSet.insert(c)
        posDigSet.insert(r + c)
        negDigSet.insert(r - c)

        dfs(r + 1)

        board[r][c] = "."
        colSet.remove(c)
        posDigSet.remove(r + c)
        negDigSet.remove(r - c)
      }
    }

    dfs(0)
    return results
  }
}
