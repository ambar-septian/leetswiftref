class Solution {
  func minReorder(_ n: Int, _ connections: [[Int]]) -> Int {
    var result = 0
    var visited = Set<Int>()
    var neighbors = [Int: [(node: Int, isOriginal: Bool)]]()

    for neighbor in connections {
      neighbors[neighbor[0], default: []].append((neighbor[1], true))
      neighbors[neighbor[1], default: []].append((neighbor[0], false))
    }

    func dfs(_ city: Int) {
      let nearestNeighbors = neighbors[city, default: []]

      for neighbor in nearestNeighbors {
        if visited.contains(neighbor.node) {
          continue
        }

        if neighbor.isOriginal {
          result += 1
        }
        visited.insert(neighbor.node)
        dfs(neighbor.node)
      }
    }

    visited.insert(0)
    dfs(0)

    return result
  }

  func minReorderOri(_ n: Int, _ connections: [[Int]]) -> Int {
    struct Edge: Hashable {
      var current: Int
      var destination: Int
    }

    var result = 0
    var edges = Set<Edge>()
    var visited = Set<Int>()
    var neighbors = [Int: [Int]]()

    for neighbor in connections {
      neighbors[neighbor[0], default: []].append(neighbor[1])
      neighbors[neighbor[1], default: []].append(neighbor[0])

      edges.insert(Edge(current: neighbor[0], destination: neighbor[1]))
    }

    func dfs(_ city: Int) {
      let nearestNeighbors = neighbors[city, default: []]

      for neighbor in nearestNeighbors {
        if visited.contains(neighbor) {
          continue
        }

        visited.insert(neighbor)
        let edge = Edge(current: neighbor, destination: city)
        if !edges.contains(edge) {
          result += 1
        }

        dfs(neighbor)
      }
    }

    visited.insert(0)
    dfs(0)

    return result
  }
}
