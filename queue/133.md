```swift
/**
 * Definition for a Node.
 * public class Node {
 *     public var val: Int
 *     public var neighbors: [Node?]
 *     public init(_ val: Int) {
 *         self.val = val
 *         self.neighbors = []
 *     }
 * }
 */

class Solution {
  func cloneGraph(_ node: Node?) -> Node? {
    guard let node else { return nil }
    var dict = [Int: Node]()

    func dfs(_ oldNode: Node?) -> Node? {
      guard let oldNode else { return nil }
      let newNode = Node(oldNode.val)
      dict[oldNode.val] = newNode

      for neighbor in oldNode.neighbors {
        guard let neighbor else { continue }
        if let node = dict[neighbor.val] {
          newNode.neighbors.append(node)
        } else {
          let neighborNode = dfs(neighbor)
          newNode.neighbors.append(neighborNode)
        }

      }

      return newNode

    }

    return dfs(node)
  }
}
```
