```swift
class Solution {
  func permute(_ nums: [Int]) -> [[Int]] {
    var results = [[Int]]()
    var path = [Int]()
    var used = Array(repeating: false, count: nums.count)

    func backTrack() {
      if path.count == nums.count {
        results.append(path)
        return
      }

      for i in 0..<nums.count {
        guard !used[i] else { continue }

        used[i] = true
        path.append(nums[i])

        backTrack()

        path.removeLast()
        used[i] = false
      }
    }

    backTrack()

    return results
  }
}
```
