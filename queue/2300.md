```swift
class Solution {
  func successfulPairs(_ spells: [Int], _ potions: [Int], _ success: Int) -> [Int] {
    var result = Array(repeating: 0, count: spells.count)
    let sortedPotions = potions.sorted()

    for i in 0..<spells.count {
      let spell = spells[i]
      var l = 0
      var r = sortedPotions.count

      while l < r {
        let m = l + (r - l) / 2
        let power = spell * sortedPotions[m]

        if power < success {
          l = m + 1
        } else {
          r = m
        }
      }

      result[i] = sortedPotions.count - l
    }

    return result
  }
}
```
