class Solution {
  func findOrder(_ numCourses: Int, _ prerequisites: [[Int]]) -> [Int] {
    var inDegree = Array(repeating: 0, count: numCourses)  // Counting the inDegree
    var results = [Int]()
    var adjList = [Int: [Int]]()

    // setup the adjacency list and indegree
    for (index, pairs) in prerequisites.enumerated() {
      let course = pairs[0]
      let prereq = pairs[1]

      adjList[prereq, default: []].append(course)
      inDegree[course] += 1
    }

    // fill the queue based on zero indegree as staring point
    var queue = [Int]()
    for (index, count) in inDegree.enumerated() {
      if count == 0 {
        queue.append(index)
      }
    }

    // similar to BFS but update the inDgree
    while !queue.isEmpty {
      let node = queue.removeFirst()
      results.append(node)

      guard let neighbors = adjList[node] else { continue }

      for neighbor in neighbors {
        inDegree[neighbor] -= 1

        if inDegree[neighbor] == 0 {
          queue.append(neighbor)
        }
      }
    }
    return results.count == numCourses ? results : []

    /*
Problem: Course Schedule II (Topological Sort / Kahn's Algorithm)
Time Complexity: O(V + E)
    - V: numCourses, E: prerequisites.count.
    - Linear traversal of all nodes and edges.
    - (Note: removeFirst() makes the BFS phase O(V^2) in Swift; use a pointer for O(V)).
Auxiliary Space: O(V + E)
    - O(V + E) to store the adjacency list.
    - O(V) for the in-degree array, queue, and topological order results.
*/
  }
}
