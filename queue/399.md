class Solution {
  func calcEquation(_ equations: [[String]], _ values: [Double], _ queries: [[String]]) -> [Double]
  {
    var adjList = [String: [(String, Double)]]()  // a -> (b, a /b)

    for (i, equation) in equations.enumerated() {
      let a = equation[0]
      let b = equation[1]

      // a -> b = 2 (comes from value)
      // b -> a = 0.5 1/ value
      let value = Double(values[i])
      adjList[a, default: []].append((b, value))
      adjList[b, default: []].append((a, 1 / value))
    }

    func bfs(_ src: String, _ target: String) -> Double {
      if adjList[src] == nil || adjList[target] == nil {
        return -1
      }

      var queue = [(String, Double)]()
      var visited = Set<String>()

      queue.append((src, 1))  // 1 is neutral value, it is safe to multiply by anything (except 0)
      visited.insert(src)

      while !queue.isEmpty {
        let (item, weight) = queue.removeFirst()
        if item == target {
          return weight
        }

        let neighbors = adjList[item, default: []]
        for neighbor in neighbors {
          if visited.contains(neighbor.0) {
            continue
          }

          visited.insert(neighbor.0)
          queue.append((neighbor.0, weight * neighbor.1))
        }
      }

      return -1

    }

    var results = [Double]()

    for query in queries {
      results.append(bfs(query[0], query[1]))
    }

    return results
  }

  /*
Problem: Evaluate Division (Graph BFS)
Time Complexity: O(M * (N + E))
    - E: Number of equations, N: Number of variables, M: Number of queries.
    - Graph construction takes O(E).
    - Each of the M queries performs a BFS of O(N + E).
    - (Assumes O(1) dequeue; Swift's removeFirst() is O(K)).
Auxiliary Space: O(N + E)
    - O(N + E) for the adjacency list.
    - O(N) for the BFS queue and visited set per query.
*/
}
