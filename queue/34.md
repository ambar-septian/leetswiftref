```swift
class Solution {
  func searchRange(_ nums: [Int], _ target: Int) -> [Int] {
    let firstItem = binarySearch(nums, target, isFindingMostLeft: true)
    let lastItem = binarySearch(nums, target, isFindingMostLeft: false)

    return [firstItem, lastItem]

    func binarySearch(_ nums: [Int], _ target: Int, isFindingMostLeft: Bool) -> Int {
      var l = 0
      var r = nums.count - 1

      var result = -1

      while l <= r {
        let m = (l + r) / 2
        if nums[m] < target {
          l = m + 1
        } else if nums[m] > target {
          r = m - 1
        } else {
          result = m
          if isFindingMostLeft {
            r = m - 1
          } else {
            l = m + 1
          }
        }
      }

      return result
    }

  }
}
```
