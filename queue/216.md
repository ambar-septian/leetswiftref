```swift
class Solution {
  func combinationSum3(_ k: Int, _ n: Int) -> [[Int]] {
    var results = [[Int]]()
    var nums = (1...9).map { $0 }
    var path = [Int]()
    var sum = 0

    func dfs(_ startIndex: Int) {
      if sum == n, path.count == k {
        results.append(path)
        return
      } else if sum > n {
        return
      } else {
        for i in startIndex..<nums.count {

          path.append(nums[i])
          sum += nums[i]
          dfs(i + 1)

          path.removeLast()
          sum -= nums[i]
        }
      }
    }

    dfs(0)

    return results
  }
}
```
