class Solution {
  func permuteUnique(_ nums: [Int]) -> [[Int]] {
    var results = [[Int]]()
    var path = [Int]()
    var used = Array(repeating: false, count: nums.count)
    var sorted = nums.sorted()
    // Recheck this logic
    func dfs() {
      if path.count == nums.count {
        results.append(path)
        return
      }

      for i in 0..<sorted.count {
        if i != 0, sorted[i] == sorted[i - 1], used[i - 1] == true {
          continue
        }

        if used[i] { continue }

        used[i] = true
        path.append(sorted[i])

        dfs()

        path.removeLast()
        used[i] = false
      }

    }

    dfs()
    return results

    // Time Complexity: O(N * N!)
    // Space Complexity: O(N * N!)
  }

  func permuteUniqueDict(_ nums: [Int]) -> [[Int]] {
    var results = [[Int]]()
    var path = [Int]()
    var cache = [Int: Int]()

    for num in nums {
      cache[num, default: 0] += 1
    }

    func dfs() {
      // Best case
      if path.count == nums.count {
        results.append(path)
        return
      }

      for c in cache where c.value != 0 {
        path.append(c.key)
        cache[c.key, default: 0] -= 1
        dfs()

        cache[c.key, default: 0] += 1
        path.removeLast()
      }
    }

    dfs()

    // Time Complexity: O(N * P)
    // Space Complexity: O(N * P)
    return results
  }
}
