```swift
class Solution {
  func isArraySpecial(_ nums: [Int]) -> Bool {
    let count = nums.count
    if count == 0 { return false }
    if count == 1 { return true }

    for i in 1..<count {
      let curr = nums[i]
      let prev = nums[i - 1]

      if (curr % 2 == 0 && prev % 2 != 0) || (curr % 2 != 0 && prev % 2 == 0) {
        continue
      } else {
        return false
      }
    }

    return true
  }
}
```
