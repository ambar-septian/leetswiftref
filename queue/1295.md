```swift
import Foundation

class Solution {
  func findNumbers(_ nums: [Int]) -> Int {
    let count = nums.count
    var result = 0

    for i in 0..<count {
      let num = nums[i]
      let value = Int(log10(Double(num))) + 1
      result += (value % 2 == 0 ? 1 : 0)
    }

    return result
  }
}
```
