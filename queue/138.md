```swift
/**
 * Definition for a Node.
 * public class Node {
 *     public var val: Int
 *     public var next: Node?
 *     public var random: Node?
 *     public init(_ val: Int) {
 *         self.val = val
 *         self.next = nil
 *    	   self.random = nil
 *     }
 * }
 */

class Solution {
  func copyRandomList(_ head: Node?) -> Node? {
    var curr = head

    while curr != nil {
      let clone = Node(curr!.val)
      clone.next = curr?.next
      curr?.next = clone
      curr = clone.next
    }

    curr = head
    while let node = curr {
      node.next?.random = node.random?.next
      curr = node.next?.next
    }

    var headCopy: Node? = Node(0)
    var copy = headCopy

    curr = head
    while let orig = curr {
      let clone = orig.next
      let nextOrig = clone?.next

      copy?.next = clone
      copy = clone

      orig.next = nextOrig
      curr = nextOrig
    }
    
    return headCopy?.next
  }

  func copyRandomListHash(_ head: Node?) -> Node? {
    var cache: [ObjectIdentifier: Node?] = [:]
    var curr = head

    var headCopy: Node? = Node(0)
    var copy = headCopy

    while curr != nil {
      copy?.next = Node(curr!.val)
      var temp = curr
      copy = copy?.next
      curr = curr?.next
      if let copy, let temp {
        cache[ObjectIdentifier(temp)] = copy
      }
    }

    curr = head
    copy = headCopy?.next
    while curr != nil {
      if let random = curr?.random, let rn = cache[ObjectIdentifier(random)] {
        copy?.random = rn
      }
      copy = copy?.next
      curr = curr?.next
    }

    return headCopy?.next
  }
}
```
