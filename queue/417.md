```swift
class Solution {
  func pacificAtlantic(_ heights: [[Int]]) -> [[Int]] {
    guard !heights.isEmpty else { return [[]] }
    let rows = heights.count
    let cols = heights[0].count
    let directions = [(0, 1), (0, -1), (1, 0), (-1, 0)]

    func encode(_ r: Int, _ c: Int) -> Int {
      r * cols + c
    }
    func decode(_ index: Int) -> [Int] {
      [index / cols, index % cols]
    }

    var queue = [Int]()

    // Build row for pacific
    for i in 0..<rows {
      queue.append(encode(i, 0))
    }

    for i in 1..<cols {
      queue.append(encode(0, i))
    }

    func bfs() -> Set<Int> {
      var oceanSet = Set<Int>()
      var visited = Set<Int>()

      while !queue.isEmpty {
        let q = queue.removeFirst()
        let grid = decode(q)
        oceanSet.insert(q)

        for dir in directions {
          let rD = grid[0] + dir.0
          let cD = grid[1] + dir.1

          if rD < 0 || cD < 0 || rD >= rows || cD >= cols {
            continue
          }

          if heights[rD][cD] < heights[grid[0]][grid[1]] {
            continue
          }

          let newGrid = encode(rD, cD)
          if visited.contains(newGrid) {
            continue
          }

          visited.insert(newGrid)
          queue.append(newGrid)
        }
      }

      return oceanSet
    }

    let pacificSet = bfs()

    // Build row for atlantic
    for i in 0..<rows {
      queue.append(encode(i, cols - 1))
    }

    for i in 0..<cols - 1 {
      queue.append(encode(rows - 1, i))
    }
    let atlanticSet = bfs()
    let finalSet = pacificSet.intersection(atlanticSet).map {
      decode($0)
    }

    return Array(finalSet)
  }

  /// Other approach use array 2D, when iteration bfs on if the array index is already true, then we can append it to result

  /*
Problem: Pacific Atlantic Water Flow (Double BFS)
Time Complexity: O(M * N)
    - M: Number of rows, N: Number of columns.
    - Two BFS passes visit each cell once.
    - O(M * N) for the final set intersection.
    - (Assumes O(1) dequeue; Swift's removeFirst() is O(K)).
Auxiliary Space: O(M * N)
    - O(M * N) to store the reachability sets (Pacific and Atlantic).
    - O(M * N) for the BFS queue and the visited set.
*/
}
```
