```swift
/// Definition for singly-linked list.
/// public class ListNode {
///     public var val: Int
///     public var next: ListNode?
///     public init() { self.val = 0; self.next = nil; }
///     public init(_ val: Int) { self.val = val; self.next = nil; }
///     public init(_ val: Int, _ next: ListNode?) { self.val = val; self.next = next; }
/// }
class Solution {
  func deleteMiddle(_ head: ListNode?) -> ListNode? {
    var slow = head
    var fast = slow?.next

    guard fast != nil else { return nil }

    while fast != nil {
      fast = fast?.next?.next

      if fast == nil {
        let next = slow?.next?.next
        slow?.next = next
      } else {
        slow = slow?.next
      }
    }

    return head

  }

  func deleteMiddleWithTwoLoops(_ head: ListNode?) -> ListNode? {
    var curr = head
    var count = 0

    while curr != nil {
      count += 1
      curr = curr?.next
    }

    let middle = count / 2
    guard middle != 0 else { return nil }

    curr = head
    var index = 0

    while curr != nil {
      if index == middle - 1 {
        let prev = curr
        let next = curr?.next?.next
        prev?.next = next
        break
      } else {
        curr = curr?.next
        index += 1
      }
    }

    return head
  }
}
```
