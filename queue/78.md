```swift
class Solution {
  func subsets(_ nums: [Int]) -> [[Int]] {
    var results = [[Int]]()
    var path = [Int]()

    func backTrack(_ i: Int) {
      if i >= nums.count {
        results.append(path)
        return
      }

      // 1. Decision Tree: Choose this path
      path.append(nums[i])
      backTrack(i + 1)

      // 2. Decision Tree: Dont choose this path
      path.removeLast()
      backTrack(i + 1)

    }

    backTrack(0)

    return results
  }

  // Time O(N * 2^N)
  // Space O(N * 2^N)

}
```
