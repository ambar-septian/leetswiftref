```swift
class Solution {
  func isCircularSentence(_ sentence: String) -> Bool {
    // transform string by space into array of spaces
    let strings = sentence.components(separatedBy: " ")
    let count = strings.count

    // Check if the sentences count 1, check the first and last char
    if count == 1 {
      let firstString = strings[0]
      if firstString.first != firstString.last {
        return false
      }
    }

    // Validate the first char of first stirng is equal to last char of last string before goes to looping
    if strings[0].first != strings[count - 1].last {
      return false
    }

    for i in 0..<count - 1 {
      // check the last character of current string is equal to first char of the next string
      let currentString = strings[i]
      let nextString = strings[i + 1]

      if currentString.last == nextString.first {
        continue
      } else {
        return false
      }
    }

    return true
  }
}
```
