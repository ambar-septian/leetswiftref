class MyCircularQueue {
  var head = 0
  var tail = 0
  var count = 0
  var items: [Int?] = []
  let maxCount: Int

  init(_ k: Int) {
    self.maxCount = k
    self.items = Array(repeating: nil, count: maxCount)
  }

  func enQueue(_ value: Int) -> Bool {
    guard count < maxCount else { return false }

    items[tail] = value
    tail = (tail + 1) % maxCount

    count += 1
    return true
  }

  func deQueue() -> Bool {
    guard count > 0 else { return false }

    items[head] = nil
    head = (head + 1) % maxCount

    count -= 1
    return true
  }

  func Front() -> Int {
    guard let value = items[head] else { return -1 }
    return value
  }

  func Rear() -> Int {
    let realTail = (tail - 1 + maxCount) % maxCount
    guard let value = items[realTail] else { return -1 }
    return value
  }

  func isEmpty() -> Bool {
    count == 0
  }

  func isFull() -> Bool {
    count == maxCount
  }
}

class Node {
  var next: Node?
  var prev: Node?
  var value: Int

  init(_ value: Int) {
    self.value = value
    self.next = nil
    self.prev = nil
  }
}

class MyCircularQueueWithNode {
  let maxCount: Int
  let left: Node
  let right: Node
  var count: Int

  init(_ k: Int) {
    self.maxCount = k
    self.left = Node(-1)
    self.right = Node(-1)
    self.count = 0

    left.next = right
    right.prev = left
  }

  func enQueue(_ value: Int) -> Bool {
    guard count < maxCount else { return false }
    var newNode = Node(value)

    var prev = right.prev
    right.prev = newNode
    newNode.next = right
    newNode.prev = prev
    prev?.next = newNode

    count += 1
    return true
  }

  func deQueue() -> Bool {
    guard count > 0 else { return false }
    var removed = left.next
    var next = removed?.next

    left.next = next
    next?.prev = left

    count -= 1
    return true
  }

  func Front() -> Int {
    guard let front = left.next else { return -1 }
    return front.value
  }

  func Rear() -> Int {
    guard let rear = right.prev else { return -1 }
    return rear.value
  }

  func isEmpty() -> Bool {
    count == 0
  }

  func isFull() -> Bool {
    count == maxCount
  }
}
