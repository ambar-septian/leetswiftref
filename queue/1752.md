```swift
class Solution {
  func check(_ nums: [Int]) -> Bool {
    let count = nums.count
    if count == 0 { return false }
    if count <= 2 { return true }
    var lowest = nums[1]
    var reset = false

    for i in 1..<count {
      let curr = nums[i]
      let prev = nums[i - 1]

      if curr >= prev {
        if !reset { continue }
        if reset, curr > nums[0] {
          return false
        }

      } else {
        if reset { return false }
        if i == count - 1 && curr > nums[0] { return false }
        reset = true
      }
    }

    return true
  }
}
```
