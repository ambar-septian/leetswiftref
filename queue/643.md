```swift
class Solution {
  func findMaxAverage(_ nums: [Int], _ k: Int) -> Double {
    var result: Double = 0
    var sum = 0
    for i in 0..<k {
      sum += nums[i]
    }

    result = Double(sum) / Double(k)

    for i in k..<nums.count {
      sum = sum - nums[i - k] + nums[i]

      result = max(result, Double(sum) / Double(k))
    }

    return result
  }
}
```
