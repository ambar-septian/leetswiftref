```swift
/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     public var val: Int
 *     public var next: ListNode?
 *     public init(_ val: Int) {
 *         self.val = val
 *         self.next = nil
 *     }
 * }
 */

class Solution {
  func deleteNode(_ node: ListNode?) {
    var curr = node
    if let val = curr?.next?.val {
      curr?.val = val
    }

    curr?.next = curr?.next?.next
  }

  func deleteNodeLoop(_ node: ListNode?) {
    var curr = node
    while curr?.next != nil {
      curr!.val = curr!.next!.val

      if curr?.next?.next == nil {
        curr?.next = nil
      } else {
        curr = curr?.next
      }

    }
  }
}
```
