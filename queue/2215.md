```swift
class Solution {
  func findDifference(_ nums1: [Int], _ nums2: [Int]) -> [[Int]] {
    var set1 = Set(nums1)
    var set2 = Set(nums2)
    var result: [[Int]] = Array(repeating: [], count: 2)

    for s1 in set1 {
      if !set2.contains(s1) {
        result[0].append(s1)
      }
    }

    for s2 in set2 {
      if !set1.contains(s2) {
        result[1].append(s2)
      }
    }

    return result
  }
}
```
