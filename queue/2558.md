import Foundation

class Solution {
  func pickGifts(_ gifts: [Int], _ k: Int) -> Int {
    let heap = Heap<Int> { l, r in l > r }
    let count = gifts.count
    var result = 0
    var left = 0
    var right = count - 1

    while left <= right {
      let a = gifts[left]

      if left == right {
        heap.insert(item: a)
        result = result + a
        break
      }

      let b = gifts[right]

      if a > b {
        heap.insert(item: a)
        heap.insert(item: b)
      } else {
        heap.insert(item: b)
        heap.insert(item: a)
      }

      left += 1
      right -= 1
      result = result + a + b
    }

    for _ in 0..<k {
      guard let top = heap.popTop() else { return 0 }
      result -= top
      let sqr = Int(Double(top).squareRoot())
      heap.insert(item: sqr)
      result += sqr
    }

    return result
  }
}

class Heap<T: Comparable> {
  typealias HeapComparator<T: Comparable> = (_ l: T, _ r: T) -> Bool
  var heap = [T]()
  var count: Int { heap.count }

  var comparator: HeapComparator<T>

  private func bubbleUp(_ index: Int) {
    let parentIndex = (index - 1) / 2

    if index <= 0 { return }
    if comparator(heap[index], heap[parentIndex]) {
      heap.swapAt(parentIndex, index)
      bubbleUp(parentIndex)
    }
  }

  private func heapify(_ index: Int) {
    var left = index * 2 + 1
    var right = index * 2 + 2

    var comp = index

    if count > left && comparator(heap[left], heap[comp]) {
      comp = left
    }

    if count > right && comparator(heap[right], heap[comp]) {
      comp = right
    }

    if comp != index {
      heap.swapAt(comp, index)
      heapify(comp)
    }
  }

  init(comparator: @escaping HeapComparator<T>) {
    self.comparator = comparator
  }

  func insert(item: T) {
    heap.append(item)
    bubbleUp(count - 1)
  }

  func getTop() -> T? {
    return heap.first
  }

  func popTop() -> T? {
    var item: T? = heap.first
    if count > 1 {
      heap[0] = heap[count - 1]
      heap.removeLast()
      heapify(0)
    } else if count == 1 {
      heap.removeLast()
    } else {
      return nil
    }

    return item
  }
}
