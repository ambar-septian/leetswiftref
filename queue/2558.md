```swift
import Collections

class Solution {
  func pickGifts(_ gifts: [Int], _ k: Int) -> Int {
    var heap = Heap<Int>()
    let count = gifts.count
    var result = 0
    var left = 0
    var right = count - 1

    while left <= right {
      let a = gifts[left]

      if left == right {
        heap.insert(a)
        result = result + a
        break
      }

      let b = gifts[right]
      heap.insert(a)
      heap.insert(b)

      left += 1
      right -= 1
      result = result + a + b
    }

    for _ in 0..<k {
      guard let top = heap.popMax() else { return 0 }
      result -= top

      let sqr = Int(Double(top).squareRoot())
      heap.insert(sqr)
      result += sqr
    }

    return result
  }
}
```
