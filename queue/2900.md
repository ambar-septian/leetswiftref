```swift
class Solution {
  func getLongestSubsequence(_ words: [String], _ groups: [Int]) -> [String] {
    var prev: Int?
    var result = [String]()

    for i in 0..<words.count {
      let word = words[i]
      let group = groups[i]

      if let prev {
        if prev != group {
          result.append(word)
        }

      } else {
        result.append(word)
      }

      prev = group
    }

    return result
  }
}
```
