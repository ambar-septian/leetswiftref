```swift
class Solution {
  func calPoints(_ operations: [String]) -> Int {
    var stack = [Int]()
    var result = 0

    for operation in operations {
      switch operation {
      case "+":
        let l1 = stack[stack.count - 1]
        let l2 = stack[stack.count - 2]
        stack.append(l1 + l2)
      case "D":
        let last = stack.last!
        let newValue = last * 2
        stack.append(newValue)
      case "C":
        stack.removeLast()
      default:
        let value = Int(operation)!
        stack.append(value)

      }
    }

    for s in stack {
      result += s
    }

    return result
  }
}
```
