```swift
class Solution {
  func numSubarrayProductLessThanK(_ nums: [Int], _ k: Int) -> Int {

    if k <= 1 { return 0 }

    var left = 0
    var total = 1
    var result = 0

    for right in 0..<nums.count {
      total = total * nums[right]

      while total >= k, left < nums.count {
        total = total / nums[left]
        left += 1
      }

      if total < k {
        result += (right - left + 1)
      }

    }

    return result
  }
}
```
