```swift
class Solution {
  func asteroidCollision(_ asteroids: [Int]) -> [Int] {
    let count = asteroids.count
    var stack: [Int] = []

    for i in 0..<count {
      var ast = asteroids[i]
      while let top = stack.last, top > 0, ast < 0 {
        if abs(ast) > abs(top) {
          stack.removeLast()
        } else if abs(top) > abs(ast) {
          ast = 0
        } else {
          stack.removeLast()
          ast = 0
        }
      }

      if ast != 0 {
        stack.append(ast)
      }

    }

    return stack
  }
}
```
