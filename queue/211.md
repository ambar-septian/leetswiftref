```swift
class Node {
  var children = [Character: Node]()
  var isEndOfWord = false
}

class WordDictionary {
  var root: Node

  init() {
    root = Node()
  }

  func addWord(_ word: String) {
    var curr = root

    for char in word {
      if let node = curr.children[char] {
        curr = node
      } else {
        curr.children[char] = Node()
        curr = curr.children[char]!
      }
    }

    curr.isEndOfWord = true
  }

  func search(_ word: String) -> Bool {
    var curr = root
    let chars = Array(word)

    func dfs(_ curr: Node?, index: Int) -> Bool {
      guard let curr else { return false }
      if index == chars.count {
        return curr.isEndOfWord
      }

      let char = chars[index]
      if char == "." {
        for child in curr.children.values {
          if dfs(child, index: index + 1) {
            return true
          }
        }

        return false
      } else if let node = curr.children[char] {
        return dfs(node, index: index + 1)
      } else {
        return false
      }

    }

    return dfs(curr, index: 0)
  }
}

/**
 * Your WordDictionary object will be instantiated and called as such:
 * let obj = WordDictionary()
 * obj.addWord(word)
 * let ret_2: Bool = obj.search(word)
 */
```
