```swift
/// Definition for a binary tree node.
/// public class TreeNode {
///     public var val: Int
///     public var left: TreeNode?
///     public var right: TreeNode?
///     public init() { self.val = 0; self.left = nil; self.right = nil; }
///     public init(_ val: Int) { self.val = val; self.left = nil; self.right = nil; }
///     public init(_ val: Int, _ left: TreeNode?, _ right: TreeNode?) {
///         self.val = val
///         self.left = left
///         self.right = right
///     }
/// }
class Solution {
  func buildTree(_ preorder: [Int], _ inorder: [Int]) -> TreeNode? {
    var hash = [Int: Int]()
    let count = inorder.count

    for i in 0..<count {
      hash[inorder[i]] = i
    }

    var preIndex = 0

    func recursive(_ inLeft: Int, _ inRight: Int) -> TreeNode? {
      if inLeft > inRight { return nil }

      let rootVal = preorder[preIndex]
      preIndex += 1
      let root = TreeNode(rootVal)

      let mid = hash[rootVal]!
      root.left = recursive(inLeft, mid - 1)
      root.right = recursive(mid + 1, inRight)

      return root
    }

    return recursive(0, inorder.count - 1)
  }

  func buildTreeTemp(_ preorder: [Int], _ inorder: [Int]) -> TreeNode? {
    guard !preorder.isEmpty, !inorder.isEmpty else {
      return nil
    }

    let root = TreeNode(preorder[0])
    guard let mid = inorder.firstIndex(where: { $0 == preorder[0] }) else {
      return nil
    }
    //    preorder[1:mid+1], inorder[0:mid]
    //    preorder[mid+1:..], inorder[mid+1:]
    if mid > 0 {
      root.left = buildTree(Array(preorder[1...mid]), Array(inorder[...mid]))
    }

    if mid + 1 < inorder.count {
      root.right = buildTree(Array(preorder[(mid + 1)...]), Array(inorder[(mid + 1)...]))
    }
    return root
  }
}
```
