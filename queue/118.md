```swift
class Solution {
  func generate(_ numRows: Int) -> [[Int]] {
    var result = Array(repeating: [Int](), count: numRows)
    guard numRows > 1 else {
      return [[1]]
    }
    result[0] = [1]

    for i in 1..<numRows {
      var rs = Array(repeating: 1, count: i + 1)
      for j in 1..<rs.count - 1 {
        rs[j] = result[i - 1][j - 1] + result[i - 1][j]
      }

      result[i] = rs
    }

    return result
  }

}
```
