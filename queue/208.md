class Node {
  var children = [Character: Node]()
  var isEndOfWord = false
}

class Trie {
  let root = Node()

  init() {

  }

  func insert(_ word: String) {
    var curr = root
    for char in word {
      if let existingNode = curr.children[char] {
        curr = existingNode
      } else {
        let node = Node()
        curr.children[char] = node
        curr = node
      }
    }

    curr.isEndOfWord = true
  }

  func search(_ word: String) -> Bool {
    var curr = root

    for char in word {
      if let existingNode = curr.children[char] {
        curr = existingNode
      } else {
        return false
      }
    }

    return curr.isEndOfWord

  }

  func startsWith(_ prefix: String) -> Bool {
    var curr = root

    for char in prefix {
      if let existingNode = curr.children[char] {
        curr = existingNode
      } else {
        return false
      }
    }

    return true
  }
}

 * let ret_2: Bool = obj.search(word)
 * let ret_3: Bool = obj.startsWith(prefix)
