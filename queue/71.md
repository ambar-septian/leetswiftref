```swift
class Solution {
  func simplifyPath(_ path: String) -> String {
    var stack = [String]()
    var current = ""

    for char in path {
      if char == "/" {
        process(&current, &stack)
      } else {
        current.append(char)
      }
    }

    process(&current, &stack)
    return "/" + stack.joined(separator: "/")
  }

  func process(_ current: inout String, _ stack: inout [String]) {
    if current == ".." {
      if !stack.isEmpty {
        stack.removeLast()
      }
    } else if !current.isEmpty, current != "." {
      stack.append(current)
    }

    current.removeAll()
  }
}
```
