```swift
class Solution {
  func combinationSum2(_ candidates: [Int], _ target: Int) -> [[Int]] {
    var results = [[Int]]()
    var path = [Int]()
    var sum = 0
    var cn = candidates.sorted()

    func dfs(_ index: Int) {
      if sum == target {
        results.append(path)
        return
      } else if sum > target {
        return
      } else {

        for i in index..<cn.count {
          let c = cn[i]
          if i > index, c == cn[i - 1] {
            continue
          }

          path.append(c)
          sum += c
          dfs(i + 1)

          path.removeLast()
          sum -= c
        }
      }
    }

    dfs(0)

    return results
  }

  /*
Problem: Combination Sum II (No reuse, has duplicates)
Time Complexity: O(2^N * N)
    - N: Number of candidates.
    - There are 2^N possible subsets.
    - For each valid subset, an O(N) copy is performed to save the result.
    - O(N log N) for the initial sort is negligible compared to 2^N.
Auxiliary Space: O(N)
    - Max depth of the recursion stack is N.
    - 'path' array and the 'cn' sorted copy take O(N).
*/
}
```
