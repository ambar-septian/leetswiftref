/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     public var val: Int
 *     public var next: ListNode?
 *     public init() { self.val = 0; self.next = nil; }
 *     public init(_ val: Int) { self.val = val; self.next = nil; }
 *     public init(_ val: Int, _ next: ListNode?) { self.val = val; self.next = next; }
 * }
 */

class Solution {
  func addTwoNumbers(_ l1: ListNode?, _ l2: ListNode?) -> ListNode? {
    guard let l1 = l1, let l2 = l2 else {
      return nil
    }

    // Calculate L1 and L2
    let total = calculate(l1: l1, l2: l2)

    // Map array of Integer into Array of ListNode. Also We need to set reference into `next` value
    let result = Solution.convert(total)
    return result
  }

  func calculate(l1: ListNode, l2: ListNode) -> [Int] {
    // We need to extract array from l1 and l2 and convert them into array
    var currentL1: ListNode? = l1
    var currentL2: ListNode? = l2
    var l1Values = [Int]()
    var l2Values = [Int]()

    while true {
      if currentL1 == nil && currentL2 == nil {
        break
      }

      if let l1Next = currentL1?.val {
        l1Values.append(l1Next)
      }

      if let l2Next = currentL2?.val {
        l2Values.append(l2Next)
      }

      currentL1 = currentL1?.next
      currentL2 = currentL2?.next
    }

    // We already have array of values, next we need to sum up each value from those array similar do math manually

    var index = 0
    var result = [Int]()
    // this variable used to tracking mod from each sum calculation
    // e.g 7 + 5 = 12
    // from result `12`, the mod would be `1` and we store `2` value into `result` array
    var mod = 0

    while index < l1Values.count || index < l2Values.count {
      var currentL1Value = 0
      var currentL2Value = 0

      if l1Values.indices.contains(index) {
        currentL1Value = l1Values[index]
      }

      if l2Values.indices.contains(index) {
        currentL2Value = l2Values[index]
      }

      // Sum the l1 and l2 along with mod
      let total = currentL1Value + currentL2Value + mod
      result.append(total % 10)

      // Track the mod
      mod = total >= 10 ? 1 : 0

      // Track the next Index
      index += 1
    }

    // If the mod is not zero until the end of looping then we need to insert the mod into first array so we can get result like this
    // e.g 500 + 500 = 1000
    if mod != 0 {
      result.append(mod)
    }

    return result
  }

  static func convert(_ values: [Int]) -> ListNode? {
    let listNodes = values.map(ListNode.init)
    for (index, child) in listNodes.enumerated() {
      if index + 1 != listNodes.count {
        child.next = listNodes[index + 1]
      }
    }

    // We only care the first ListNode, ignore the rest of ListNodes
    return listNodes.first
  }

}
