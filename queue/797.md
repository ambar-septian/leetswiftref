```swift
class Solution {
  func allPathsSourceTarget(_ graph: [[Int]]) -> [[Int]] {
    var result = [[Int]]()
    var list = [0]
    let n = graph.count - 1

    func dfs(_ i: Int) {
      if i == n {
        result.append(list)
        return
      }

      let node = graph[i]
      for neighbor in node {
        list.append(neighbor)
        dfs(neighbor)

        list.removeLast()
      }
    }

    dfs(0)
    return result
  }

  /*
Problem: All Paths From Source to Target (DAG Backtracking)
Time Complexity: O(N * 2^N)
    - N: Number of nodes.
    - There are up to 2^N possible paths in a DAG.
    - Each valid path takes O(N) time to copy into the result array.
Auxiliary Space: O(N)
    - O(N) for the recursion stack and the 'list' array.
    - Does not count the space for the final 'result' output.
*/
}
```
