```swift
class Solution {
  func twoSum(_ nums: [Int], _ target: Int) -> [Int] {
    var dict = [Int: Int]()  // Key: value from nums, Value: Index

    for i in 0..<nums.count {
      let targetIndex = target - nums[i]
      if let existingIndex = dict[targetIndex] {
        return [existingIndex, i]
      } else {
        dict[nums[i]] = i
      }
    }

    return []
  }
}
```
