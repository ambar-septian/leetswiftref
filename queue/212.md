class Solution {
  func findWords(_ board: [[Character]], _ words: [String]) -> [String] {
    var root = TrieNode()
    buildTrie(root, words)

    let rows = board.count
    let cols = board[0].count
    var board = board

    var results = [String]()

    func dfs(_ r: Int, _ c: Int, _ node: TrieNode?) {
      guard let node else { return }

      // Pruning after found the end of wrod
      if let word = node.word {
        node.word = nil
        results.append(word)
      }

      if r < 0 || c < 0 || r >= rows || c >= cols {
        return
      }

      guard let nextNode = node.children[board[r][c]] else {
        return
      }

      let temp = board[r][c]
      board[r][c] = "#"

      dfs(r + 1, c, nextNode)
      dfs(r - 1, c, nextNode)
      dfs(r, c + 1, nextNode)
      dfs(r, c - 1, nextNode)

      board[r][c] = temp

      if nextNode.word == nil, nextNode.children.isEmpty {
        node.children[board[r][c]] = nil
      }

    }

    for r in 0..<rows {
      for c in 0..<cols {
        dfs(r, c, root)
      }
    }

    return results
  }

  func buildTrie(_ root: TrieNode, _ words: [String]) {
    var curr = root

    for word in words {
      curr = root

      for char in word {
        if let exist = curr.children[char] {
          curr = exist
        } else {
          let node = TrieNode()
          curr.children[char] = node
          curr = node
        }
      }

      curr.word = word
    }
  }
}

class TrieNode {
  var children = [Character: TrieNode]()
  var word: String?

  func printTrie(_ node: TrieNode, prefix: String = "", level: Int = 0) {
    let indent = String(repeating: "  ", count: level)

    for (char, child) in node.children.sorted(by: { $0.key < $1.key }) {
      let wordMarker = child.word != nil ? " (END)" : ""
      print("\(indent)└─ \(char)\(wordMarker)")
      printTrie(child, prefix: prefix + String(char), level: level + 1)
    }
  }
}
