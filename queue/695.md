```swift
class Solution {
  func maxAreaOfIsland(_ grid: [[Int]]) -> Int {
    guard !grid.isEmpty else { return 0 }
    var grid = grid
    let rows = grid.count
    let cols = grid[0].count
    let directions = [(0, 1), (0, -1), (1, 0), (-1, 0)]
    var results = 0

    func dfs(_ r: Int, _ c: Int) -> Int {
      if r < 0 || c < 0 || r >= rows || c >= cols || grid[r][c] == 0 {
        return 0
      }

      // mark as visited
      grid[r][c] = 0
      var count = 1

      for dir in directions {
        let rDir = r + dir.0
        let cDir = c + dir.1
        count += dfs(rDir, cDir)
      }

      return count
    }

    for r in 0..<rows {
      for c in 0..<cols {
        if grid[r][c] == 1 {
          results = max(results, dfs(r, c))
        }
      }
    }

    return results
  }

  /*
Problem: All Paths From Source to Target (DAG Backtracking)
Time Complexity: O(N * 2^N)
    - N: Number of nodes.
    - There are up to 2^N possible paths in a DAG.
    - Each valid path takes O(N) time to copy into the result array.
Auxiliary Space: O(N)
    - O(N) for the recursion stack and the 'list' array.
    - Does not count the space for the final 'result' output.
*/
}
```
