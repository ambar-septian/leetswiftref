```swift
class Node {
  let key: Int
  var value: Int

  var next: Node?
  var prev: Node?

  init(key: Int, value: Int) {
    self.key = key
    self.value = value
  }
}

class LRUCache {
  let capacity: Int
  let left: Node
  let right: Node
  var cache: [Int: Node]

  init(_ capacity: Int) {
    self.capacity = capacity
    self.cache = [:]

    self.left = Node(key: -1, value: -1)
    self.right = Node(key: -1, value: -1)

    left.next = right
    right.prev = left
  }

  func get(_ key: Int) -> Int {
    guard let node = cache[key] else {
      return -1
    }

    remove(node)
    insert(node)

    return node.value
  }

  func put(_ key: Int, _ value: Int) {
    if let node = cache[key] {
      // Update the value
      node.value = value

      // Move to recent node
      remove(node)
      insert(node)
    } else {
      // Create the node and assign into cache and the listNode
      let node = Node(key: key, value: value)
      cache[key] = node
      insert(node)

      // Remove the least node if it is exceed
      if cache.count > capacity, let leastNode = left.next {
        cache[leastNode.key] = nil
        remove(leastNode)
      }
    }
  }

  private func insert(_ newNode: Node) {
    let prev = right.prev
    // Update the previous link
    prev?.next = newNode
    newNode.prev = prev

    // Update the next link (rightNode)
    newNode.next = right
    right.prev = newNode
  }

  private func remove(_ deletedNode: Node) {
    let prev = deletedNode.prev
    let next = deletedNode.next

    prev?.next = next
    next?.prev = prev
  }
}

/**
 * Your LRUCache object will be instantiated and called as such:
 * let obj = LRUCache(capacity)
 * let ret_1: Int = obj.get(key)
 * obj.put(key, value)
 */
```
