```swift
class Solution {
  func isPalindrome(_ s: String) -> Bool {
    let array: [Character] = Array(s)
    let count = array.count
    var left = 0
    var right = count - 1

    while left != right && left < count && right >= 0 {
      let a = Character(array[left].lowercased())
      let b = Character(array[right].lowercased())

      if a == b {
        left += 1
        right -= 1
      } else {
        var cont = false
        let asciiA = a.asciiValue ?? 0
        if !((asciiA >= 97 && asciiA <= 122) || (asciiA >= 48 && asciiA <= 57)) {
          left += 1
          cont = true
        }

        let asciiB = b.asciiValue ?? 0
        if !((asciiB >= 97 && asciiB <= 122) || (asciiB >= 48 && asciiB <= 57)) {
          right -= 1
          cont = true
        }

        if cont { continue }
        return false
      }
    }

    return true
  }
}
```
