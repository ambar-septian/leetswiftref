```swift
class Solution {
  func isPalindrome(_ x: Int) -> Bool {
    if x < 0 {
      return false
    }

    let array = String(x).map(String.init)
    let count = array.count
    var pointer = 0
    let middleIndex = count / 2
    while pointer < middleIndex {
      if array[pointer] == array[count - pointer - 1] {
        pointer += 1
        continue
      } else {
        return false
      }
    }

    return true
  }
}
```
