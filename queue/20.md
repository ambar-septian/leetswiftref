```swift
class Solution {
  func isValid(_ s: String) -> Bool {
    var stack = [String]()
    let count = s.count
    guard count > 1, count % 2 == 0 else {
      return false
    }

    for char in s {
      switch char {
      case "{":
        stack.append("{")
      case "}":
        if !stack.isEmpty && stack.last == "{" {
          stack.removeLast()
        } else {
          stack.append("}")
        }

      case "[":
        stack.append("[")
      case "]":
        if !stack.isEmpty && stack.last == "[" {
          stack.removeLast()
        } else {
          stack.append("]")
        }

      case "(":
        stack.append("(")
      case ")":
        if !stack.isEmpty && stack.last == "(" {
          stack.removeLast()
        } else {
          stack.append(")")
        }
      default:
        break
      }
    }

    return stack.isEmpty

  }
}
```
