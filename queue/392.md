```swift
class Solution {
  func isSubsequence(_ s: String, _ t: String) -> Bool {
    let schars = s.map { $0 }
    let tchars = t.map { $0 }
    let tcount = tchars.count
    let scount = schars.count

    var i = 0
    var j = 0

    while j < tcount, i < scount {
      if schars[i] == tchars[j] {
        i += 1
      }

      j += 1
    }

    return i == scount
  }

  func isSubsequenceStack(_ s: String, _ t: String) -> Bool {
    var schars = s.map { $0 }
    let tchars = t.map { $0 }

    for i in 0..<tchars.count {
      if let schar = schars.first, schar == tchars[i] {
        schars.removeFirst()
      }
    }

    return schars.isEmpty
  }
}
```
