```swift
import Collections

class Solution {
  func maxScore(_ nums1: [Int], _ nums2: [Int], _ k: Int) -> Int {
    var pairs = Array(zip(nums1, nums2)).sorted(by: { $0.1 > $1.1 })  // sort descending by nums2
    var score = 0
    var heap = Heap<Int>()
    var sum = 0

    for (i, pair) in pairs.enumerated() {
      sum += pair.0
      heap.insert(pair.0)

      if heap.count > k, let minVal = heap.popMin() {
        sum -= minVal
      }

      if heap.count == k {
        score = max(score, sum * pair.1)
      }

    }

    return score

    /*
Problem: Maximum Subsequence Score (Sorting + Min-Heap)
Time Complexity: O(N log N)
    - N: Number of elements in nums1/nums2.
    - Sorting pairs takes O(N log N).
    - Iterating through N pairs with O(log k) heap operations takes O(N log k).
Auxiliary Space: O(N)
    - O(N) to store the zipped and sorted pairs.
    - O(k) for the min-heap to track the largest elements from nums1.
*/
  }
}
```
