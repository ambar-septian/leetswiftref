```swift
class Solution {
  func wordPattern(_ pattern: String, _ s: String) -> Bool {
    let chars = pattern.map { String($0) }
    let words = s.components(separatedBy: " ")
    var charSet = [String: String]()
    var wordSet = [String: String]()

    guard chars.count == words.count else { return false }

    for i in 0..<chars.count {
      let char = chars[i]
      let word = words[i]

      if charSet[char] == nil, wordSet[word] == nil {
        charSet[char] = word
        wordSet[word] = char
      } else if charSet[char] == word && wordSet[word] == char {
        continue
      } else {
        return false
      }
    }

    return true
  }
}
```
