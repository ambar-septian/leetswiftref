```swift
class Solution {
  func finalPrices(_ prices: [Int]) -> [Int] {
    var stack = [(value: Int, index: Int)]()
    var result = prices

    for i in 0..<prices.count {
      while let top = stack.last, prices[i] <= top.value {
        stack.removeLast()
        result[top.index] = top.value - prices[i]
      }

      stack.append((value: prices[i], index: i))
    }

    return result
  }

}
```
