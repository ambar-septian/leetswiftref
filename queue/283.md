---

```swift
class Solution {
  func moveZeroes(_ nums: inout [Int]) {
    let count = nums.count
 
    var insertPosition = 0
    for i in 0..<count {
      if nums[i] != 0 {
        nums.swapAt(i, insertPosition)
        insertPosition += 1
      }
    }
  }
}
```
