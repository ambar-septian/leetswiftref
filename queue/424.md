```swift
class Solution {
  func characterReplacement(_ s: String, _ k: Int) -> Int {
    guard !s.isEmpty else { return 0 }

    let chars = Array(s)
    var left = 0
    var right = 0
    var dict = [Character: Int]()
    var result = 0
    var dominant = 0

    while right < chars.count {
      dict[chars[right], default: 0] += 1
      dominant = max(dominant, dict[chars[right], default: 0])
  
      while (right - left + 1) - dominant > k {
        dict[chars[left], default: 0] -= 1
        left += 1
      }

      result = max(right - left + 1, result)
      right += 1
    }

    // ABABC

    return result
  }
}
```
