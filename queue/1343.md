```swift
class Solution {
  func numOfSubarrays(_ arr: [Int], _ k: Int, _ threshold: Int) -> Int {
    var sum = 0

    for i in 0..<k {
      sum += arr[i]
    }

    var avg = sum / k
    var result = avg >= threshold ? 1 : 0

    for i in k..<arr.count {
      sum = sum - arr[i - k] + arr[i]
      avg = sum / k
      if avg >= threshold {
        result += 1
      }
    }

    return result
  }
}
```
