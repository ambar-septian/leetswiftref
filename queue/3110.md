```swift
class Solution {
  func scoreOfString(_ s: String) -> Int {
    let arr = s.map { $0 }
    var result = 0
    for i in 1..<arr.count {
      result = result + abs(Int(arr[i].asciiValue!) - Int(arr[i - 1].asciiValue!))
    }

    return result
  }
}
```
