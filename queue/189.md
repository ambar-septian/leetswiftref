```swift
class Solution {
  func rotate(_ nums: inout [Int], _ k: Int) {
    let c = k % nums.count  // This is going to the middle index when the rotation is finished
    var l = 0
    var r = nums.count - 1

    func updatePointers() {
      l = l + 1
      r = r - 1
    }

    // Reverse the array
    while l < r {
      nums.swapAt(l, r)
      updatePointers()
    }

    // Rotate the left side
    l = 0
    r = c - 1
    while l < r {
      nums.swapAt(l, r)
      updatePointers()
    }

    // Rotate the right side
    l = c
    r = nums.count - 1

    while l < r {
      nums.swapAt(l, r)
      updatePointers()
    }
  }
}
```
