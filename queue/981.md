import Foundation

class TimeMap {

  struct Entry {
    let value: String
    let timestamp: Int
  }

  var entries: [String: [Entry]]

  init() {
    entries = [:]
  }

  func set(_ key: String, _ value: String, _ timestamp: Int) {
    entries[key, default: []].append(Entry(value: value, timestamp: timestamp))
  }

  func get(_ key: String, _ timestamp: Int) -> String {
    guard let existingEntries = entries[key] else {
      return ""
    }

    // Find the timestamp
    var l = 0
    var r = existingEntries.count - 1
    var result = ""

    while l <= r {
      let mid = (l + r) / 2
      if existingEntries[mid].timestamp <= timestamp {
        result = existingEntries[mid].value
        l = mid + 1
      } else {
        r = mid - 1
      }
    }

    return result
  }
}

/**
 * Your TimeMap object will be instantiated and called as such:
 * let obj = TimeMap()
 * obj.set(key, value, timestamp)
 * let ret_2: String = obj.get(key, timestamp)
