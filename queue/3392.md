```swift
class Solution {
  func countSubarrays(_ nums: [Int]) -> Int {
    let count = nums.count
    var result = 0

    for i in 0..<count - 2 {
      let first = nums[i]
      let middle = nums[i + 1]
      let last = nums[i + 2]

      if Float(middle) / 2 == Float(first + last) {

        result += 1
      }
    }

    return result
  }
}
```
