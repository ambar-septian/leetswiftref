```swift
class Solution {
  func largestRectangleArea(_ heights: [Int]) -> Int {
    var maxArea = 0
    var stack = [(index: Int, height: Int)]()

    for i in 0..<heights.count {
      var start = i  // This is used to track the previous index or how long the height can be extended backwards
      let height = heights[i]

      // Check if the last item is higher than current item
      // IF true we keep pop it and calculate the height by check of the current index - latest height index

      while let top = stack.last, top.height > height {
        let (index, height) = stack.removeLast()
        maxArea = max(maxArea, height * (i - index))
        start = index  // Update how far the index can be extended backwards
      }

      // Keep append the item to stack
      stack.append((start, height))
    }

    for item in stack {
      maxArea = max(maxArea, item.height * (heights.count - item.index))
    }

    return maxArea
  }
}
```
