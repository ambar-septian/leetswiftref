class Solution {
  func reverseWords(_ s: String) -> String {
    var words: [Substring] = []
    let str = s
    var i = str.startIndex

    while i < str.endIndex {
      while i < str.endIndex, str[i] == " " {
        i = str.index(after: i)
      }

      if i == str.endIndex { break }
      var j = i
      while j < str.endIndex, str[j] != " " {
        j = str.index(after: j)
      }

      words.append(str[i..<j])

      i = j
    }

    return words.reversed().joined(separator: " ")
  }

  func reverseWordsBad(_ s: String) -> String {
    var l = 0
    var r = 0
    var result = ""
    let count = s.count

    var i = s.startIndex
    while i < s.endIndex {
      defer { i = s.index(after: i) }

      if s[i] != " " && i.encodedOffset != count - 1 {
        r += 1
      } else {
        let start = s.index(s.startIndex, offsetBy: l)
        let endIndex = i.encodedOffset == count - 1 && s[i] != " " ? r + 1 : r
        let end = s.index(s.startIndex, offsetBy: endIndex)

        let substring = String(s[start..<end])
        if substring == " " || substring == "" {
          r += 1
          l = r
          continue
        }

        if result == "" {
          result = "\(String(substring))"
        } else {
          result = "\(String(substring)) \(result)"
        }

        r += 1
        l = r
      }
    }

    return result
  }

  func simpleReverseWords(_ s: String) -> String {
    return Array(s.components(separatedBy: " ").filter { $0 != "" }.reversed()).joined(
      separator: " ")
  }
}
