```swift
class Solution {
  func replaceWords(_ dictionary: [String], _ sentence: String) -> String {
    let root: Node = Node()
    var current: Node?
    var newSentences = sentence.components(separatedBy: " ")

    for (index, dict) in dictionary.enumerated() {
      current = root

      for char in dict {
        if current?.children[char] == nil {
          current?.children[char] = Node()
        }

        current = current?.children[char]
      }

      current?.isEndOfWord = true
      current?.word = dict

    }

    for (index, word) in newSentences.enumerated() {
      current = root
      for char in word {
        if current?.isEndOfWord == true {
          newSentences[index] = current?.word ?? ""
          break
        } else if current?.children[char] != nil {
          current = current?.children[char]
        } else {
          break
        }
      }
    }

    return newSentences.joined(separator: " ")
  }

  class Node {
    var children = [Character: Node]()
    var isEndOfWord = false
    var word = ""

    func debugPrint(prefix: String = "", depth: Int = 0) {
      let indent = String(repeating: "  ", count: depth)

      for key in children.keys.sorted() {
        print("\(indent)\(key)")
        children[key]?.debugPrint(prefix: prefix + String(key), depth: depth + 1)
      }
    }
  }
}
```
