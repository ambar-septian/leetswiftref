```swift
class Solution {
  func removeDuplicates(_ s: String) -> String {
    let chars = Array(s)
    var stack = [Character]()
    for char in chars {
      var c: Character? = char
      while let top = stack.last, top == c {
        c = nil
        stack.removeLast()
      }

      if let c {
        stack.append(c)
      }
    }

    return String(stack)
  }
}
```
