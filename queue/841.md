class Solution {
  func canVisitAllRooms(_ rooms: [[Int]]) -> Bool {
    var visited = Set<Int>()

    func dfs(_ i: Int) {
      guard !visited.contains(i) else {
        return
      }

      visited.insert(i)
      let keys = rooms[i]

      for key in keys {
        dfs(key)
      }
    }
    dfs(0)
    return visited.count == rooms.count
  }

  /*
Problem: Keys and Rooms (Optimized DFS)
Time Complexity: O(N + K)
    - N: Number of rooms.
    - K: Total number of keys.
    - Every room and every key is processed exactly once.
Auxiliary Space: O(N)
    - O(N) for the 'visited' set and the recursion stack.
    - Memory efficient as it uses the input 'rooms' array directly.
*/
}
