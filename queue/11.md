```swift
class Solution {
  func maxArea(_ height: [Int]) -> Int {
    var left = 0
    var right = height.count - 1
    var total = 0

    while left != right {
      let minHeight = min(height[left], height[right])
      let width = right - left
      total = max(width * minHeight, total)

      if height[left] > height[right] {
        right -= 1
      } else {
        left += 1
      }
    }

    return total
  }
}
```
